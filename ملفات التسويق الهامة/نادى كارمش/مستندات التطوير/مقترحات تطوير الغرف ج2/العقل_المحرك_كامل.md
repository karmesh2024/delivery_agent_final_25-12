<a name="philosophy"></a>
## 2️⃣ فلسفة العقل المحرك

### المبدأ الأساسي

> **"الذكاء الخفي - النظام يفهمك بدون أن تشعر"**

```
التقليدي (❌):
┌─────────────────┐
│ املأ الاستبيان  │
│ 50 سؤال        │
│ 20 دقيقة       │
└─────────────────┘
النتيجة: المستخدم يهرب

الذكي (✅):
┌─────────────────┐
│ تصفح طبيعي     │
│ تفاعل عادي     │
│ النظام يتعلم   │
└─────────────────┘
النتيجة: بروفايل دقيق بدون جهد
```

### المكونات الثلاثة

```
1. الجمع (Collection)
   ↓
   جمع البيانات من كل تفاعل
   
2. التحليل (Analysis)
   ↓
   فهم الأنماط والتفضيلات
   
3. التطبيق (Application)
   ↓
   اقتراحات ذكية + مكافآت تلقائية
```

// services/dataCollection.service.ts

interface BehaviorEvent {
  customerId: string;
  roomId: string;
  actionType: 'view' | 'like' | 'comment' | 'share' | 'create';
  targetType: 'post' | 'comment' | 'member';
  targetId: string;
  timeSpent?: number;
  metadata?: any;
}

class DataCollectionService {
  
  /**
   * تسجيل سلوك المستخدم
   */
  async logBehavior(event: BehaviorEvent) {
    // 1. حفظ في behavior logs
    await prisma.roomBehaviorLogs.create({
      data: {
        roomId: event.roomId,
        customerId: event.customerId,
        actionType: event.actionType,
        targetType: event.targetType,
        targetId: event.targetId,
        timeSpent: event.timeSpent,
        actionMetadata: event.metadata
      }
    });
    
    // 2. تحديث إحصائيات العضوية
    await this.updateMembershipStats(event);
    
    // 3. تحديث البروفايل
    await this.updateProfileFromBehavior(event);
  }
  
  /**
   * تحديث إحصائيات العضوية
   */
  private async updateMembershipStats(event: BehaviorEvent) {
    const updates: any = {
      lastVisitAt: new Date()
    };
    
    // تحديث حسب نوع الحدث
    if (event.actionType === 'view') {
      updates.postsViewed = { increment: 1 };
    } else if (event.actionType === 'like' || event.actionType === 'comment') {
      updates.postsEngaged = { increment: 1 };
    } else if (event.actionType === 'create') {
      updates.postsCreated = { increment: 1 };
    }
    
    // إضافة الوقت
    if (event.timeSpent) {
      updates.totalTimeSpent = { 
        increment: Math.round(event.timeSpent / 60) 
      };
    }
    
    await prisma.roomMemberships.update({
      where: {
        roomId_customerId: {
          roomId: event.roomId,
          customerId: event.customerId
        }
      },
      data: updates
    });
  }

   * تحديث البروفايل من السلوك
   */

    // جلب البروفايل
    const profile = await prisma.customerProfiles.findUnique({
      where: { customerId: event.customerId }
    });
    
    if (!profile) return;
    
    // استنتاج الاهتمامات من السلوك
    const insights = await this.analyzeB ehavior(event);

    await prisma.customerProfiles.update({
      where: { customerId: event.customerId },
      data: {
        inferredInterests: {
          ...profile.inferredInterests,
          ...insights.interests
        },
        behaviorPatterns: {
          ...profile.behaviorPatterns,
          ...insights.patterns
        },
        dataPointsCollected: { increment: 1 },
        lastProfilingUpdate: new Date()
      }
    });
  }
  
  /**
   * تحليل السلوك واستنتاج الأنماط
   */
  private async analyzeBehavior(event: BehaviorEvent) {
    const insights: any = {
      interests: {},
      patterns: {}
    };
    
    // استنتاج الاهتمامات
    if (event.actionType === 'like' || event.actionType === 'comment') {
      // جلب موضوع المنشور
      const post = await this.getPostDetails(event.targetId);
      if (post && post.tags) {
        post.tags.forEach(tag => {
          insights.interests[tag] = (insights.interests[tag] || 0) + 1;
        });
      }
    }
    
    // استنتاج الأنماط السلوكية
    if (event.timeSpent && event.timeSpent > 30) {
      insights.patterns.deepReader = true;
    }
    
    if (event.actionType === 'comment') {
      insights.patterns.activeParticipant = true;
    }
    
    return insights;
  }
  
  private async getPostDetails(postId: string) {
    // جلب تفاصيل المنشور من جدول المنشورات
    return await prisma.posts.findUnique({
      where: { id: postId },
      select: { tags: true, category: true }
    });
  }

// services/compatibility.service.ts

interface CompatibilityResult {
  overallScore: number;
  personalityScore: number;
  interestsScore: number;
  behaviorScore: number;
  breakdown: {
    sharedInterests: string[];
    personalitySimilarity: any;
    compatibilityFactors: string[];
  };
}

class CompatibilityEngine {
  
  /**
   * حساب التوافق بين عضوين
   */
  async calculateCompatibility(
    memberA: string,
    memberB: string
  ): Promise<CompatibilityResult> {
    
    // 1. جلب البروفايلات
    const [profileA, profileB] = await Promise.all([
      prisma.customerProfiles.findUnique({ where: { customerId: memberA } }),
      prisma.customerProfiles.findUnique({ where: { customerId: memberB } })
    ]);
    
    if (!profileA || !profileB) {
      throw new Error('Profile not found');
    }
    
    // 2. حساب توافق الاهتمامات (40%)
    const interestsScore = this.calculateInterestsCompatibility(
      profileA.interests,
      profileB.interests
    );
    
    // 3. حساب توافق الشخصية (40%)
    const personalityScore = this.calculatePersonalityCompatibility(
      profileA,
      profileB
    );
    
    // 4. حساب توافق السلوك (20%)
    const behaviorScore = await this.calculateBehaviorCompatibility(
      memberA,
      memberB
    );
    
    // 5. الدرجة الإجمالية
    const overallScore = 
      (interestsScore * 0.4) +
      (personalityScore * 0.4) +
      (behaviorScore * 0.2);
    
    // 6. التفاصيل
    const breakdown = await this.generateBreakdown(profileA, profileB);
    
    // 7. حفظ النتيجة
    await this.saveCompatibility(memberA, memberB, {
      overallScore,
      personalityScore,
      interestsScore,
      behaviorScore,
      ...breakdown
    });
    
    return {
      overallScore: Math.round(overallScore),
      personalityScore: Math.round(personalityScore),
      interestsScore: Math.round(interestsScore),
      behaviorScore: Math.round(behaviorScore),
      breakdown
    };
  }
  
  /**
   * حساب توافق الاهتمامات
   */
  private calculateInterestsCompatibility(
    interestsA: string[],
    interestsB: string[]
  ): number {
    if (!interestsA?.length || !interestsB?.length) return 0;
    
    // الاهتمامات المشتركة
    const shared = interestsA.filter(i => interestsB.includes(i));
    
    // النسبة
    const unionSize = new Set([...interestsA, ...interestsB]).size;
    const similarity = (shared.length / unionSize) * 100;
    
    return similarity;
  }
  
  /**
   * حساب توافق الشخصية (Big Five)
   */
  private calculatePersonalityCompatibility(
    profileA: any,
    profileB: any
  ): number {
    const dimensions = [
      'openness_score',
      'conscientiousness_score',
      'extraversion_score',
      'agreeableness_score',
      'neuroticism_score'
    ];
    
    let totalSimilarity = 0;
    let validDimensions = 0;
    
    dimensions.forEach(dim => {
      const scoreA = profileA[dim];
      const scoreB = profileB[dim];
      
      if (scoreA && scoreB) {
        // حساب المسافة (كلما قلت = أفضل)
        const distance = Math.abs(scoreA - scoreB);
        
        // تحويل لنسبة توافق (100 = متطابق، 0 = متعاكس)
        const similarity = 100 - distance;
        
        totalSimilarity += similarity;
        validDimensions++;
      }
    });
    
    return validDimensions > 0 ? totalSimilarity / validDimensions : 50;
  }
  
  /**
   * حساب توافق السلوك
   */
  private async calculateBehaviorCompatibility(
    memberA: string,
    memberB: string
  ): number {
    // جلب الغرف المشتركة
    const sharedRooms = await this.getSharedRooms(memberA, memberB);
    
    if (sharedRooms.length === 0) return 50; // محايد
    
    let totalScore = 0;
    
    for (const room of sharedRooms) {
      // جلب سلوك كل منهما في الغرفة
      const [behaviorA, behaviorB] = await Promise.all([
        this.getRoomBehavior(memberA, room.id),
        this.getRoomBehavior(memberB, room.id)
      ]);
      
      // مقارنة أنماط السلوك
      const similarity = this.compareBehaviorPatterns(behaviorA, behaviorB);
      totalScore += similarity;
    }
    
    return totalScore / sharedRooms.length;
  }
  
  /**
   * جلب الغرف المشتركة
   */
  private async getSharedRooms(memberA: string, memberB: string) {
    const roomsA = await prisma.roomMemberships.findMany({
      where: { customerId: memberA, isActive: true },
      select: { roomId: true, room: true }
    });
    
    const roomsB = await prisma.roomMemberships.findMany({
      where: { customerId: memberB, isActive: true },
      select: { roomId: true }
    });
    
    const roomIdsB = new Set(roomsB.map(r => r.roomId));
    
    return roomsA.filter(r => roomIdsB.has(r.roomId));
  }
  
  /**
   * جلب سلوك في غرفة
   */
  private async getRoomBehavior(customerId: string, roomId: string) {
    return await prisma.roomMemberships.findUnique({
      where: {
        roomId_customerId: { roomId, customerId }
      }
    });
  }
  
  /**
   * مقارنة أنماط السلوك
   */
  private compareBehaviorPatterns(behaviorA: any, behaviorB: any): number {
    if (!behaviorA || !behaviorB) return 50;
    
    let score = 0;
    
    // نمط المشاركة (creator vs viewer)
    const creatorA = behaviorA.postsCreated > 5;
    const creatorB = behaviorB.postsCreated > 5;
    if (creatorA === creatorB) score += 25;
    
    // نمط التفاعل
    const engagementA = behaviorA.postsEngaged / Math.max(behaviorA.postsViewed, 1);
    const engagementB = behaviorB.postsEngaged / Math.max(behaviorB.postsViewed, 1);
    const engagementDiff = Math.abs(engagementA - engagementB);
    score += (1 - engagementDiff) * 25;
    
    // الوقت المقضي
    const timeA = behaviorA.totalTimeSpent;
    const timeB = behaviorB.totalTimeSpent;
    const timeDiff = Math.abs(timeA - timeB) / Math.max(timeA, timeB, 1);
    score += (1 - timeDiff) * 25;
    
    // نشاط مستمر
    const activeA = behaviorA.visitCount > 10;
    const activeB = behaviorB.visitCount > 10;
    if (activeA === activeB) score += 25;
    
    return score;
  }
  
  /**
   * إنشاء تفاصيل التوافق
   */
  private async generateBreakdown(profileA: any, profileB: any) {
    // الاهتمامات المشتركة
    const sharedInterests = profileA.interests.filter(
      (i: string) => profileB.interests.includes(i)
    );
    
    // توافق الشخصية
    const personalitySimilarity = {
      openness: this.getDimensionSimilarity(
        profileA.openness_score,
        profileB.openness_score
      ),
      extraversion: this.getDimensionSimilarity(
        profileA.extraversion_score,
        profileB.extraversion_score
      )
    };
    
    // عوامل التوافق
    const compatibilityFactors = [];
    if (sharedInterests.length > 3) {
      compatibilityFactors.push('اهتمامات مشتركة كثيرة');
    }
    if (personalitySimilarity.extraversion > 80) {
      compatibilityFactors.push('أنماط شخصية متشابهة');
    }
    
    return {
      sharedInterests,
      personalitySimilarity,
      compatibilityFactors
    };
  }
  
  private getDimensionSimilarity(scoreA: number, scoreB: number): number {
    if (!scoreA || !scoreB) return 50;
    return 100 - Math.abs(scoreA - scoreB);
  }
  
  /**
   * حفظ نتيجة التوافق
   */
  private async saveCompatibility(
    memberA: string,
    memberB: string,
    result: any
  ) {
    await prisma.memberCompatibility.upsert({
      where: {
        memberA_id_memberB_id: { member_a_id: memberA, member_b_id: memberB }
      },
      create: {
        member_a_id: memberA,
        member_b_id: memberB,
        overall_score: result.overallScore,
        personality_score: result.personalityScore,
        interests_score: result.interestsScore,
        behavior_score: result.behaviorScore,
        shared_interests: result.breakdown.sharedInterests,
        personality_similarity: result.breakdown.personalitySimilarity,
        compatibility_factors: result.breakdown.compatibilityFactors,
        calculated_at: new Date(),
        expires_at: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000) // أسبوع
      },
      update: {
        overall_score: result.overallScore,
        personality_score: result.personalityScore,
        interests_score: result.interestsScore,
        behavior_score: result.behaviorScore,
        calculated_at: new Date()
      }
    });
  }
}

class CircleStrengthService {
  
  /**
   * حساب قوة الدائرة
   */
  async calculateCircleStrength(circleId: string): Promise<number> {
    // 1. جلب الدائرة والأعضاء
    const circle = await prisma.circles.findUnique({
      where: { id: circleId },
      include: {
        members: {
          where: { memberStatus: 'active' },
          include: { customer: true }
        }
      }
    });
    
    if (!circle) return 0;
    
    let strength = 0;
    
    // عدد الأعضاء (30%)
    strength += Math.min((circle.totalMembers / 10) * 30, 30);
    
    // متوسط التوافق (40%)
    const avgCompatibility = await this.getAverageCompatibility(circleId);
    strength += (avgCompatibility / 100) * 40;
    
    // النشاط (30%)
    const activityScore = await this.getActivityScore(circleId);
    strength += activityScore * 0.3;
    
    // تحديث في قاعدة البيانات
    await prisma.circles.update({
      where: { id: circleId },
      data: {
        circleStrength: Math.round(strength),
        avgCompatibilityScore: avgCompatibility
      }
    });
    
    // فحص المكافآت القابلة للفتح
    await this.checkUnlockableRewards(circleId, Math.round(strength));
    
    return Math.round(strength);
  }
  
  /**
   * حساب متوسط التوافق
   */
  private async getAverageCompatibility(circleId: string): Promise<number> {
    const members = await prisma.circleMembers.findMany({
      where: { circleId, memberStatus: 'active' },
      select: { compatibilityScore: true }
    });
    
    if (members.length === 0) return 0;
    
    const sum = members.reduce((acc, m) => acc + (m.compatibilityScore || 0), 0);
    return sum / members.length;
  }
  
  /**
   * حساب نشاط الدائرة
   */
  private async getActivityScore(circleId: string): Promise<number> {
    const circle = await prisma.circles.findUnique({
      where: { id: circleId },
      select: {
        totalMessages: true,
        totalResources: true,
        updatedAt: true
      }
    });
    
    if (!circle) return 0;
    
    let score = 0;
    
    // الرسائل
    score += Math.min((circle.totalMessages / 100) * 50, 50);
    
    // الموارد
    score += Math.min((circle.totalResources / 20) * 30, 30);
    
    // النشاط الأخير (خلال 7 أيام = 20 نقطة)
    const daysSinceUpdate = Math.floor(
      (Date.now() - circle.updatedAt.getTime()) / (1000 * 60 * 60 * 24)
    );
    if (daysSinceUpdate <= 7) {
      score += 20 - (daysSinceUpdate * 2);
    }
    
    return Math.min(score, 100);
  }
  
  /**
   * فحص المكافآت القابلة للفتح
   */
  private async checkUnlockableRewards(circleId: string, strength: number) {
    const circle = await prisma.circles.findUnique({
      where: { id: circleId },
      include: {
        rewards: {
          where: { isUnlocked: false }
        },
        members: {
          where: { memberStatus: 'active' }
        }
      }
    });
    
    if (!circle) return;
    
    for (const reward of circle.rewards) {
      let shouldUnlock = false;
      
      // فحص الشرط
      if (reward.unlockCondition === 'members_count') {
        shouldUnlock = circle.totalMembers >= reward.unlockConditionValue;
      } else if (reward.unlockCondition === 'circle_strength') {
        shouldUnlock = strength >= reward.unlockConditionValue;
      }
      
      if (shouldUnlock) {
        // فتح المكافأة
        await prisma.circleRewards.update({
          where: { id: reward.id },
          data: {
            isUnlocked: true,
            unlockedAt: new Date()
          }
        });
        
        // إشعار الأعضاء
        await this.notifyRewardUnlocked(circle, reward);
      }
    }
  }
  
  /**
   * إشعار بفتح مكافأة
   */
  private async notifyRewardUnlocked(circle: any, reward: any) {
    const message = `🎉 تم فتح ميزة جديدة: ${reward.rewardName}`;
    
    // إرسال إشعار لكل الأعضاء
    for (const member of circle.members) {
      await prisma.notifications.create({
        data: {
          customerId: member.customerId,
          type: 'REWARD_UNLOCKED',
          title: 'ميزة جديدة!',
          message,
          metadata: {
            circleId: circle.id,
            rewardType: reward.rewardType
          }
        }
      });
    }
  }
}

// services/externalInvitations.service.ts

class ExternalInvitationsService {
  
  /**
   * إنشاء دعوة خارجية
   */
  async createInvitation(params: {
    inviterId: string;
    circleId: string;
    inviteePhone: string;
    inviteeName?: string;
    customMessage?: string;
  }) {
    
    // 1. توليد كود فريد
    const invitationCode = this.generateInvitationCode();
    
    // 2. إنشاء الدعوة
    const invitation = await prisma.externalInvitations.create({
      data: {
        inviterId: params.inviterId,
        circleId: params.circleId,
        inviteePhone: params.inviteePhone,
        inviteeName: params.inviteeName,
        invitationCode,
        invitationMessage: params.customMessage,
        invitationMethod: 'whatsapp',
        status: 'pending',
        expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000) // 7 أيام
      }
    });
    
    // 3. جلب معلومات الدائرة والمنشئ
    const [circle, inviter] = await Promise.all([
      prisma.circles.findUnique({ where: { id: params.circleId } }),
      prisma.customers.findUnique({ where: { id: params.inviterId } })
    ]);
    
    // 4. إنشاء رابط الدعوة
    const invitationLink = this.generateInvitationLink(invitationCode);
    
    // 5. إنشاء رسالة WhatsApp
    const whatsappMessage = this.generateWhatsAppMessage({
      inviterName: inviter?.full_name || 'صديقك',
      circleName: circle?.name || 'دائرة جديدة',
      customMessage: params.customMessage,
      invitationLink
    });
    
    // 6. إنشاء رابط WhatsApp
    const whatsappUrl = this.generateWhatsAppUrl(
      params.inviteePhone,
      whatsappMessage
    );
    
    return {
      invitation,
      whatsappUrl,
      invitationLink
    };
  }
  
  /**
   * توليد كود دعوة فريد
   */
  private generateInvitationCode(): string {
    const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
    let code = '';
    for (let i = 0; i < 8; i++) {
      code += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return code;
  }
  
  /**
   * توليد رابط الدعوة
   */
  private generateInvitationLink(code: string): string {
    return `https://zoon.zone/join/${code}`;
  }
  
  /**
   * توليد رسالة WhatsApp
   */
  private generateWhatsAppMessage(params: {
    inviterName: string;
    circleName: string;
    customMessage?: string;
    invitationLink: string;
  }): string {
    let message = `🌟 مرحباً!\n\n`;
    message += `${params.inviterName} يدعوك للانضمام إلى "${params.circleName}" في نادي زوون ZONE\n\n`;
    
    if (params.customMessage) {
      message += `رسالة خاصة: ${params.customMessage}\n\n`;
    }
    
    message += `انضم الآن:\n${params.invitationLink}\n\n`;
    message += `نادي زوون - حيث تجد من يفهمك 💙`;
    
    return message;
  }
  
  /**
   * توليد رابط WhatsApp
   */
  private generateWhatsAppUrl(phone: string, message: string): string {
    // تنظيف رقم الهاتف
    const cleanPhone = phone.replace(/\D/g, '');
    
    // ترميز الرسالة
    const encodedMessage = encodeURIComponent(message);
    
    return `https://wa.me/${cleanPhone}?text=${encodedMessage}`;
  }
  
  /**
   * قبول الدعوة
   */
  async acceptInvitation(code: string, customerId: string) {
    // 1. جلب الدعوة
    const invitation = await prisma.externalInvitations.findUnique({
      where: { invitationCode: code }
    });
    
    if (!invitation) {
      throw new Error('دعوة غير موجودة');
    }
    
    if (invitation.status !== 'pending') {
      throw new Error('الدعوة تم قبولها مسبقاً أو انتهت صلاحيتها');
    }
    
    if (new Date() > invitation.expiresAt) {
      throw new Error('انتهت صلاحية الدعوة');
    }
    
    // 2. إضافة العضو للدائرة
    const member = await prisma.circleMembers.create({
      data: {
        circleId: invitation.circleId,
        customerId,
        memberStatus: 'active',
        compatibilityScore: 0 // سيُحسب لاحقاً
      }
    });
    
    // 3. تحديث الدعوة
    await prisma.externalInvitations.update({
      where: { id: invitation.id },
      data: {
        status: 'accepted',
        acceptedAt: new Date()
      }
    });
    
    // 4. تحديث عدد الأعضاء
    await prisma.circles.update({
      where: { id: invitation.circleId },
      data: {
        totalMembers: { increment: 1 }
      }
    });
    
    // 5. حساب التوافق
    const compatibilityEngine = new CompatibilityEngine();
    const inviterCompatibility = await compatibilityEngine.calculateCompatibility(
      invitation.inviterId,
      customerId
    );
    
    // 6. تحديث نسبة التوافق
    await prisma.circleMembers.update({
      where: { id: member.id },
      data: {
        compatibilityScore: inviterCompatibility.overallScore
      }
    });
    
    // 7. إشعار المدعي
    await prisma.notifications.create({
      data: {
        customerId: invitation.inviterId,
        type: 'INVITATION_ACCEPTED',
        title: 'قبول دعوة',
        message: `قبل ${invitation.inviteeName || 'صديقك'} دعوتك للانضمام للدائرة`,
        metadata: {
          circleId: invitation.circleId,
          newMemberId: customerId
        }
      }
    });
    
    // 8. إعادة حساب قوة الدائرة
    const strengthService = new CircleStrengthService();
    await strengthService.calculateCircleStrength(invitation.circleId);
    
    return {
      circle: await prisma.circles.findUnique({
        where: { id: invitation.circleId }
      }),
      compatibility: inviterCompatibility
    };
  }
  
  /**
   * تتبع النقرات على الدعوة
   */
  async trackInvitationClick(code: string) {
    await prisma.externalInvitations.update({
      where: { invitationCode: code },
      data: {
        clicks: { increment: 1 },
        lastClickedAt: new Date()
      }
    });
  }
}

class AutoPriceUpdater {
    public function updatePrices() {
        $materials = StockExchange::where('auto_update_enabled', true)->get();
        
        foreach ($materials as $material) {
            // 1. حساب إجمالي المخزون
            $totalStock = $this->calculateTotalStock($material->category_id, $material->subcategory_id);
            
            // 2. حساب متوسط أسعار الموردين
            $avgSupplierPrice = $this->calculateAvgSupplierPrice($material->category_id, $material->subcategory_id);
            
            // 3. تحديد مستوى الطلب
            $demandLevel = $this->calculateDemandLevel($totalStock);
            
            // 4. حساب السعر الجديد
            $newPrice = $this->calculateNewPrice($material->base_price, $totalStock, $avgSupplierPrice, $demandLevel);
            
            // 5. تحديث البيانات
            $material->update([
                'current_price' => $newPrice,
                'total_stock' => $totalStock,
                'avg_supplier_price' => $avgSupplierPrice,
                'demand_level' => $demandLevel,
                'price_change_percentage' => (($newPrice - $material->base_price) / $material->base_price) * 100
            ]);
        }
    }
    
    private function calculateDemandLevel($totalStock) {
        if ($totalStock < 200) return 'critical';
        if ($totalStock < 500) return 'high';
        if ($totalStock > 1000) return 'low';
        return 'normal';
    }
}

        // تنبيهات المخزون المنخفض

function calculateNewPrice(basePrice, totalStock, avgSupplierPrice, demandLevel) {
    let priceMultiplier = 1;
    
    // تأثير مستوى الطلب (المخزون)
    switch (demandLevel) {
        case 'critical':
            priceMultiplier *= 1.15; // زيادة 15%
            break;
        case 'high':
            priceMultiplier *= 1.08; // زيادة 8%
            break;
        case 'low':
            priceMultiplier *= 0.95; // تخفيض 5%
            break;
        default:
            priceMultiplier *= 1.02; // زيادة طفيفة 2%
    }
    
    // تأثير أسعار الموردين
    let supplierInfluence = 0.3; // 30% تأثير من أسعار الموردين
    let marketInfluence = 0.7;   // 70% تأثير من السوق
    
    let newPrice = (basePrice * priceMultiplier * marketInfluence) + 
                   (avgSupplierPrice * supplierInfluence);
    
    // تحديد حدود التغيير (لا يزيد أو ينقص أكثر من 20%)
    let maxPrice = basePrice * 1.20;
    let minPrice = basePrice * 0.80;
    
    return Math.max(minPrice, Math.min(maxPrice, newPrice));
}

SELECT 
    se.category_id,
    se.subcategory_id,
    se.current_price,
    AVG(spo.proposed_price) as avg_supplier_price,
    (se.current_price - AVG(spo.proposed_price)) as profit_margin,
    SUM(wi.quantity) as total_available_stock
FROM stock_exchange_advanced se
LEFT JOIN supplier_price_offers spo ON se.category_id = spo.category_id 
    AND se.subcategory_id = spo.subcategory_id
LEFT JOIN warehouse_inventory wi ON se.category_id = wi.category_id 
    AND se.subcategory_id = wi.subcategory_id
GROUP BY se.category_id, se.subcategory_id
ORDER BY profit_margin DESC;

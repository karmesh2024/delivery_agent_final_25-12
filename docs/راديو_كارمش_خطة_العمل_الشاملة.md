### الفكرة الأساسية

**راديو كارمش** هو راديو صوتي مباشر داخل نادي Scope Zone يُبث عبر:
- 📱 **تطبيق الموبايل** (Flutter - تطبيق كارمش الحالي)
- 🌐 **موقع الإنترنت** (Next.js - Dashboard الحالي)
- 🎧 **داخل النادي** (سماعات/شاشات في فروع النادي)

### المحتوى المقترح

| نوع المحتوى | الوصف | التكرار |
|------------|-------|---------|
| **بث مباشر** | مقابلات مع ضيوف (دكتور، صاحب محل، خبير بيئي) | أسبوعي |

## 🎯 الأهداف الاستراتيجية

### 1. زيادة التفاعل مع الأعضاء
- **المشكلة**: الأعضاء غير متفاعلين بشكل كافي
- **الحل**: محتوى صوتي جذاب يسمعه الأعضاء أثناء اليوم
- **النتيجة المتوقعة**: زيادة التفاعل بنسبة 40-60%

### 2. إيرادات من الرعاة
- **المشكلة**: الحاجة لمصادر إيراد إضافية
- **الحل**: رعاية البث المباشر من الشركاء
- **النتيجة المتوقعة**: إيراد شهري من الرعاة ($200-500)

### 3. بناء الولاء
- **المشكلة**: الحاجة لتقوية ارتباط الأعضاء بالنادي
- **الحل**: محتوى حصري صوتي يربط الأعضاء بالنادي
- **النتيجة المتوقعة**: زيادة معدل الاحتفاظ بنسبة 30%

### 4. نظام نقاط تفاعلي
- **المشكلة**: الحاجة لزيادة استخدام نقاط النادي
- **الحل**: منح نقاط للمستمعين الحاضرين أثناء البث
- **النتيجة المتوقعة**: زيادة استخدام نقاط النادي بنسبة 50%

### 5. تسويق الشركاء
- **المشكلة**: الحاجة لتوفير قيمة للشركاء
- **الحل**: ظهور الشركاء في البث كرعاة
- **النتيجة المتوقعة**: جذب شركاء جدد وزيادة رضا الشركاء الحاليين

```bash

# 3. تثبيت أدوات أساسية
```

```sql
-- تحديث club_activities لدعم البث
ALTER TABLE club_activities

COMMENT ON TABLE radio_listeners IS 'تتبع المستمعين الحاليين للبث (لنظام النقاط)';

-- جدول سجلات البث
CREATE TABLE IF NOT EXISTS radio_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  activity_id UUID NOT NULL REFERENCES club_activities(id) ON DELETE CASCADE,
  started_at TIMESTAMPTZ DEFAULT NOW(),
  ended_at TIMESTAMPTZ,
  total_listeners INTEGER DEFAULT 0,
  peak_listeners INTEGER DEFAULT 0,
  total_duration_minutes INTEGER DEFAULT 0,
  partner_id UUID REFERENCES club_partners(id),
  status VARCHAR(20) DEFAULT 'active' CHECK (status IN ('active', 'completed', 'cancelled')),

CREATE INDEX idx_radio_sessions_activity_id ON radio_sessions(activity_id);
CREATE INDEX idx_radio_sessions_status ON radio_sessions(status);

COMMENT ON TABLE radio_sessions IS 'سجلات جلسات البث المباشر';

-- RLS Policies
ALTER TABLE radio_listeners ENABLE ROW LEVEL SECURITY;
ALTER TABLE radio_sessions ENABLE ROW LEVEL SECURITY;

-- Users can read their own listening records
CREATE POLICY "Users can read own listening records"
ON radio_listeners FOR SELECT
USING (user_id = auth.uid());

-- Admins can read all
CREATE POLICY "Admins can read all listening records"
ON radio_listeners FOR SELECT

    WHERE user_id = auth.uid()
    AND is_active = true
```

#### الخطوة 3.2: إنشاء Radio Service

**ملف جديد: `src/domains/club-zone/services/clubRadioService.ts`**

```typescript
/**
 * Club Radio Service
 * إدارة البث الصوتي
 */

import { supabase } from '@/lib/supabase';
import { ClubActivity } from '../types';

export interface RadioStream {
  id: string;
  title: string;
  stream_url: string;
  listen_url: string;
  current_listeners: number;
  max_listeners: number;
  is_live: boolean;
  partner_id?: string;
  started_at?: string;
}

export interface RadioListener {
  id: string;
  user_id: string;
  activity_id: string;
  started_listening_at: string;
  last_active_at: string;
  duration_minutes: number;
  points_earned: number;
  is_active: boolean;
}

export const clubRadioService = {
  /**
   * الحصول على البث الحالي
   */
  async getCurrentStream(): Promise<RadioStream | null> {
    const { data, error } = await supabase
      .from('club_activities')
      .select(`
        *,
        club_partners (
          id,
          company_name,
          logo_url
        )
      `)
      .eq('activity_type', 'radio_stream')
      .eq('is_active', true)
      .eq('is_live', true)
      .single();

    if (error || !data) return null;

    return {
      id: data.id,
      title: data.title,
      stream_url: data.stream_url || '',
      listen_url: data.listen_url || '',
      current_listeners: data.current_listeners || 0,
      max_listeners: data.max_listeners || 1000,
      is_live: data.is_live || false,
      partner_id: data.partner_id,
      started_at: data.scheduled_at,
    };
  },

  /**
   * بدء الاستماع (لتتبع النقاط)
   */
  async startListening(userId: string, activityId: string): Promise<void> {
    const { error } = await supabase
      .from('radio_listeners')
      .upsert({
        user_id: userId,
        activity_id: activityId,
        started_listening_at: new Date().toISOString(),
        last_active_at: new Date().toISOString(),
        is_active: true,
      }, {
        onConflict: 'user_id,activity_id',
        ignoreDuplicates: false,
      });

    if (error) throw error;
  },

  /**
   * تحديث نشاط المستمع (heartbeat)
   */
  async updateListenerActivity(userId: string, activityId: string): Promise<void> {
    const { error } = await supabase
      .from('radio_listeners')
      .update({ 
        last_active_at: new Date().toISOString(),
        is_active: true,
      })
      .eq('user_id', userId)
      .eq('activity_id', activityId);

    if (error) throw error;
  },

  /**
   * إيقاف الاستماع
   */
  async stopListening(userId: string, activityId: string): Promise<void> {
    const { data: listener } = await supabase
      .from('radio_listeners')
      .select('started_listening_at')
      .eq('user_id', userId)
      .eq('activity_id', activityId)
      .single();

    if (!listener) return;

    // حساب المدة بالدقائق
    const durationMinutes = Math.floor(
      (new Date().getTime() - new Date(listener.started_listening_at).getTime()) / 60000
    );

    // تحديث السجل
    const { error } = await supabase
      .from('radio_listeners')
      .update({ 
        is_active: false,
        duration_minutes: durationMinutes,
      })
      .eq('user_id', userId)
      .eq('activity_id', activityId);

    if (error) throw error;

    // منح النقاط (نقطة لكل دقيقة)
    if (durationMinutes > 0) {
      await supabase.rpc('update_club_points_wallet', {
        p_user_id: userId,
        p_points: durationMinutes,
        p_transaction_type: 'EARNED',
        p_reason: 'radio_listening',
        p_source: 'radio_stream',
        p_description: `نقاط الاستماع للراديو - ${durationMinutes} دقيقة`,
      });
    }
  },

  /**
   * الحصول على عدد المستمعين الحاليين
   */
  async getCurrentListeners(activityId: string): Promise<number> {
    const { count, error } = await supabase
      .from('radio_listeners')
      .select('*', { count: 'exact', head: true })
      .eq('activity_id', activityId)
      .eq('is_active', true)
      .gt('last_active_at', new Date(Date.now() - 5 * 60 * 1000).toISOString()); // نشط آخر 5 دقائق

    if (error) throw error;
    return count || 0;
  },

  /**
   * منح نقاط للمستمعين (Background Job)
   */
  async awardPointsToListeners(activityId: string, pointsPerMinute: number = 1): Promise<void> {
    // يتم استدعاؤه من Background Job كل 5 دقائق
    
    // الحصول على المستمعين النشطين
    const { data: listeners, error } = await supabase
      .from('radio_listeners')
      .select('*')
      .eq('activity_id', activityId)
      .eq('is_active', true)
      .gt('last_active_at', new Date(Date.now() - 5 * 60 * 1000).toISOString());

    if (error || !listeners) return;

    for (const listener of listeners) {
      // حساب المدة بالدقائق
      const minutesListened = Math.floor(
        (new Date().getTime() - new Date(listener.started_listening_at).getTime()) / 60000
      );
      
      // حساب النقاط المستحقة
      const pointsToAward = (minutesListened * pointsPerMinute) - (listener.points_earned || 0);

      if (pointsToAward > 0) {
        // منح النقاط
        await supabase.rpc('update_club_points_wallet', {
          p_user_id: listener.user_id,
          p_points: pointsToAward,
          p_transaction_type: 'EARNED',
          p_reason: 'radio_listening',
          p_source: 'radio_stream',
          p_description: `نقاط الاستماع للراديو - ${minutesListened} دقيقة`,
        });

        // تحديث السجل
        await supabase
          .from('radio_listeners')
          .update({ 
            points_earned: minutesListened * pointsPerMinute,
            duration_minutes: minutesListened,
          })
          .eq('id', listener.id);
      }
    }
  },

  /**
   * إنشاء/بدء بث جديد
   */
  async createStream(streamData: {
    title: string;
    scheduled_at: string;
    partner_id?: string;
    stream_url: string;
    listen_url: string;
    max_listeners?: number;
  }): Promise<ClubActivity> {
    const { data, error } = await supabase
      .from('club_activities')
      .insert([{
        activity_type: 'radio_stream',
        title: streamData.title,
        scheduled_at: streamData.scheduled_at,
        partner_id: streamData.partner_id,
        stream_url: streamData.stream_url,
        listen_url: streamData.listen_url,
        max_listeners: streamData.max_listeners || 1000,
        is_active: true,
        is_live: false, // سيتم تفعيله يدوياً
      }])
      .select()
      .single();

    if (error) throw error;
    return data as ClubActivity;
  },

  /**
   * تفعيل البث المباشر
   */
  async startLiveStream(activityId: string): Promise<void> {
    const { error } = await supabase
      .from('club_activities')
      .update({ 
        is_live: true,
        current_listeners: 0,
      })
      .eq('id', activityId);

    if (error) throw error;

    // إنشاء سجل للبث
    await supabase
      .from('radio_sessions')
      .insert([{
        activity_id: activityId,
        status: 'active',
        started_at: new Date().toISOString(),
      }]);
  },

  /**
   * إيقاف البث المباشر
   */
  async stopLiveStream(activityId: string): Promise<void> {
    // تحديث حالة البث
    const { error } = await supabase
      .from('club_activities')
      .update({ 
        is_live: false,
      })
      .eq('id', activityId);

    if (error) throw error;

    // إنهاء سجل البث
    const { data: session } = await supabase
      .from('radio_sessions')
      .select('id, started_at')
      .eq('activity_id', activityId)
      .eq('status', 'active')
      .order('started_at', { ascending: false })
      .limit(1)
      .single();

    if (session) {
      const durationMinutes = Math.floor(
        (new Date().getTime() - new Date(session.started_at).getTime()) / 60000
      );

      await supabase
        .from('radio_sessions')
        .update({
          status: 'completed',
          ended_at: new Date().toISOString(),
          total_duration_minutes: durationMinutes,
        })
        .eq('id', session.id);
    }
  },
};
```

```dart
// ملف: lib/features/club_zone/services/radio_service.dart

import 'package:just_audio/just_audio.dart';
import 'package:http/http.dart' as http;
import 'dart:convert';

class RadioService {
  final AudioPlayer _player = AudioPlayer();
  String? _currentStreamUrl;
  String? _userId;
  String? _activityId;
  DateTime? _startedAt;
  
  // Timer لتحديث النشاط
  Timer? _activityTimer;

  Future<void> startListening({
    required String streamUrl,
    required String userId,
    required String activityId,
  }) async {
    try {
      _currentStreamUrl = streamUrl;
      _userId = userId;
      _activityId = activityId;
      _startedAt = DateTime.now();

      // تشغيل البث
      await _player.setUrl(streamUrl);
      await _player.play();

      // إرسال طلب بدء الاستماع
      await _sendStartListeningRequest(userId, activityId);

      // بدء Timer لتحديث النشاط كل 30 ثانية
      _activityTimer = Timer.periodic(Duration(seconds: 30), (timer) {
        _updateListenerActivity(userId, activityId);
      });
    } catch (e) {
      print('Error starting radio: $e');
      rethrow;
    }
  }

  Future<void> stopListening() async {
    try {
      await _player.stop();
      
      if (_userId != null && _activityId != null) {
        await _sendStopListeningRequest(_userId!, _activityId!);
      }

      _activityTimer?.cancel();
      _activityTimer = null;
      _currentStreamUrl = null;
      _userId = null;
      _activityId = null;
      _startedAt = null;
    } catch (e) {
      print('Error stopping radio: $e');
    }
  }

  Future<void> _sendStartListeningRequest(String userId, String activityId) async {
    final response = await http.post(
      Uri.parse('https://your-api.com/api/club-zone/radio/start-listening'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({
        'user_id': userId,
        'activity_id': activityId,
      }),
    );

    if (response.statusCode != 200) {
      throw Exception('Failed to start listening');
    }
  }

  Future<void> _updateListenerActivity(String userId, String activityId) async {
    try {
      await http.post(
        Uri.parse('https://your-api.com/api/club-zone/radio/update-activity'),
        headers: {'Content-Type': 'application/json'},
        body: jsonEncode({
          'user_id': userId,
          'activity_id': activityId,
        }),
      );
    } catch (e) {
      print('Error updating activity: $e');
    }
  }

  Future<void> _sendStopListeningRequest(String userId, String activityId) async {
    await http.post(
      Uri.parse('https://your-api.com/api/club-zone/radio/stop-listening'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({
        'user_id': userId,
        'activity_id': activityId,
      }),
    );
  }

  bool get isPlaying => _player.playing;
  Stream<Duration> get positionStream => _player.positionStream;
  Stream<Duration> get durationStream => _player.durationStream;

  Future<void> pause() => _player.pause();
  Future<void> play() => _player.play();
}
```

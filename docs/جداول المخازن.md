-- جدول المخازن (warehouses)
-- هذا الجدول يمثل المستودعات الفعلية أو مواقع التخزين.
-- يعتمد على جدول المناطق (regions) لتحديد الموقع الجغرافي.
CREATE TABLE warehouses (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL UNIQUE,
    location TEXT,
    region_id INTEGER REFERENCES regions(id),
    capacity DECIMAL(10,2),
    current_stock DECIMAL(10,2) DEFAULT 0,
    manager_name TEXT,
    contact_phone TEXT,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- سياسة الحماية للمخازن (RLS for warehouses)
ALTER TABLE warehouses ENABLE ROW LEVEL SECURITY;
CREATE POLICY "المخازن مرئية للمستخدمين المصرح لهم" ON warehouses FOR SELECT USING (true);
CREATE POLICY "المخازن قابلة للتعديل من المسؤولين" ON warehouses FOR INSERT WITH CHECK (true);
CREATE POLICY "المخازن قابلة للتحديث من المسؤولين" ON warehouses FOR UPDATE USING (true);

-- جدول مخزون المواد (warehouse_inventory)
-- هذا الجدول يسجل الكميات الحالية لكل منتج/مادة في مستودع معين.
-- يعتمد على جداول المنتجات العامة (public.categories, public.subcategories, public.waste_data_admin).
CREATE TABLE warehouse_inventory (
  id SERIAL PRIMARY KEY,
  warehouse_id INTEGER REFERENCES warehouses(id),
  category_id UUID REFERENCES public.categories(id),
  subcategory_id UUID REFERENCES public.subcategories(id),
  product_id UUID REFERENCES public.waste_data_admin(id),
  quantity DECIMAL(10,2) DEFAULT 0,
  unit TEXT,
  min_level DECIMAL(10,2),
  max_level DECIMAL(10,2),
  last_updated TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  UNIQUE(warehouse_id, product_id)
);

-- سياسة الحماية لمخزون المواد (RLS for warehouse_inventory)
ALTER TABLE warehouse_inventory ENABLE ROW LEVEL SECURITY;
CREATE POLICY "مخزون المواد مرئي للمستخدمين المصرح لهم" ON warehouse_inventory FOR SELECT USING (true);
CREATE POLICY "مخزون المواد قابل للتعديل من المسؤولين" ON warehouse_inventory FOR INSERT WITH CHECK (true);
CREATE POLICY "مخزون المواد قابل للتحديث من المسؤولين" ON warehouse_inventory FOR UPDATE USING (true);

-- جدول حركات المخزون (inventory_movements)
CREATE TABLE inventory_movements (
  id serial not null,
  warehouse_id integer null,
  product_id uuid null,
  category_id uuid null,
  subcategory_id uuid null,
  movement_type text null,
  quantity numeric(10, 2) not null,
  unit text null,
  price numeric(10, 2) null,
  source_type text null,
  source_id text null,
  notes text null,
  created_at timestamp with time zone null default now(),
  created_by uuid null,
  constraint inventory_movements_pkey primary key (id),
  constraint inventory_movements_category_id_fkey foreign KEY (category_id) references categories (id),
  constraint inventory_movements_product_id_fkey foreign KEY (product_id) references waste_data_admin (id),
  constraint inventory_movements_subcategory_id_fkey foreign KEY (subcategory_id) references subcategories (id),
  constraint inventory_movements_warehouse_id_fkey foreign KEY (warehouse_id) references warehouses (id),
  constraint inventory_movements_movement_type_check check (
    (
      movement_type = any (array['in'::text, 'out'::text])
    )
  )
) TABLESPACE pg_default;

-- دالة تحديث المخزون عند تسجيل حركة جديدة (Function: update_inventory_on_movement)
-- هذه الدالة تقوم بتحديث جدول warehouse_inventory وإجمالي المخزون في جدول warehouses
-- عند إدراج أو تعديل حركة في جدول inventory_movements.
CREATE OR REPLACE FUNCTION update_inventory_on_movement()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.movement_type = 'in' THEN
    -- إضافة للمخزون
    INSERT INTO warehouse_inventory 
      (warehouse_id, product_id, category_id, subcategory_id, quantity, unit)
    VALUES 
      (NEW.warehouse_id, NEW.product_id, NEW.category_id, NEW.subcategory_id, NEW.quantity, NEW.unit)
    ON CONFLICT (warehouse_id, product_id) 
    DO UPDATE SET
      quantity = warehouse_inventory.quantity + NEW.quantity,
      last_updated = NOW();
  ELSIF NEW.movement_type = 'out' THEN
    -- خصم من المخزون
    UPDATE warehouse_inventory
    SET 
      quantity = GREATEST(0, quantity - NEW.quantity),
      last_updated = NOW()
    WHERE 
      warehouse_id = NEW.warehouse_id AND product_id = NEW.product_id;
  END IF;
  
  -- تحديث إجمالي المخزون في المخزن
  UPDATE warehouses
  SET 
    current_stock = (SELECT COALESCE(SUM(quantity), 0) FROM warehouse_inventory WHERE warehouse_id = NEW.warehouse_id),
    updated_at = NOW()
  WHERE id = NEW.warehouse_id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- إنشاء الـ Trigger (trg_update_inventory)
-- هذا المحفز يقوم بتشغيل دالة update_inventory_on_movement بعد كل إدراج في جدول inventory_movements.
CREATE TRIGGER trg_update_inventory
AFTER INSERT ON inventory_movements
FOR EACH ROW
EXECUTE FUNCTION update_inventory_on_movement();

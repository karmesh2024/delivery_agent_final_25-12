    region_id INTEGER REFERENCES regions(id),
    supplier_type_id INTEGER REFERENCES supplier_types(id) NOT NULL,

    contact_person_ar TEXT,
    contact_phone TEXT,
    contact_phone_secondary TEXT,
    email TEXT,
    website TEXT,
    
    -- العنوان

    address_ar TEXT,
    city TEXT,
    postal_code TEXT,

    tax_number TEXT,

-- جدول التفاصيل المالية للموردين (محدث ومحسن)
CREATE TABLE supplier_financial_details (
    id SERIAL PRIMARY KEY,
    supplier_id INTEGER REFERENCES suppliers(id) ON DELETE CASCADE UNIQUE, -- جعلها UNIQUE لكل مورد
    
    -- التفاصيل المالية الأساسية
    preferred_payment_type_id INTEGER REFERENCES payment_types(id),
    secondary_payment_type_id INTEGER REFERENCES payment_types(id),
    credit_limit DECIMAL(15,2) DEFAULT 0, -- الحد الائتماني المتاح للمورد
    current_balance DECIMAL(15,2) DEFAULT 0,
    outstanding_amount DECIMAL(15,2) DEFAULT 0,
    
    -- شروط الدفع
    payment_terms_days INTEGER DEFAULT 30,
    installment_months INTEGER, -- عدد الأشهر في حالة الأقساط
    installment_amount DECIMAL(15,2), -- قيمة القسط الشهري
    discount_percentage DECIMAL(5,2) DEFAULT 0, -- نسبة خصم للدفع المبكر
    late_fee_percentage DECIMAL(5,2) DEFAULT 0, -- نسبة غرامة التأخير
    
    -- البيانات المصرفية
    bank_name TEXT,
    bank_account_number TEXT,
    iban TEXT,
    swift_code TEXT,
    account_holder_name TEXT,
    
    -- تفاصيل الضمانات المالية
    bank_guarantee_amount DECIMAL(15,2) DEFAULT 0,
    guarantee_expiry_date DATE,
    insurance_amount DECIMAL(15,2) DEFAULT 0,
    insurance_expiry_date DATE,
    
    -- إعدادات التعامل المالي
    requires_advance_payment BOOLEAN DEFAULT FALSE,
    advance_payment_percentage DECIMAL(5,2) DEFAULT 0,
    auto_deduct_enabled BOOLEAN DEFAULT FALSE, -- خصم تلقائي من الحساب
    
    -- تواريخ مالية مهمة
    last_payment_date DATE,
    next_payment_due_date DATE,
    account_opening_date DATE DEFAULT CURRENT_DATE,
    
    -- ملاحظات مالية
    financial_notes TEXT,
    risk_assessment TEXT, -- تقييم المخاطر المالية
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- جدول المواد/المنتجات التي يقدمها المورد (جدول مدمج ومحسن)

    id SERIAL PRIMARY KEY,
    supplier_id INTEGER REFERENCES suppliers(id) ON DELETE CASCADE,
    product_id UUID REFERENCES waste_data_admin(id) ON DELETE CASCADE,
    
    -- تفاصيل السعر والكمية
    purchase_price DECIMAL(10,2) NOT NULL,
    minimum_quantity DECIMAL(10,2) DEFAULT 0,
    maximum_quantity DECIMAL(10,2),
    preferred_quantity DECIMAL(10,2),
    price_unit TEXT NOT NULL,
    
    -- تفاصيل التعامل المالي للمادة/المنتج
    payment_type_id INTEGER REFERENCES payment_types(id),
    payment_terms_days INTEGER DEFAULT 30,
    requires_deposit BOOLEAN DEFAULT FALSE,
    deposit_percentage DECIMAL(5,2) DEFAULT 0,
    
    -- مواصفات الجودة والشروط
    quality_specifications TEXT,
    purity_percentage DECIMAL(5,2), -- نسبة النقاء للمادة
    moisture_content_max DECIMAL(5,2), -- نسبة الرطوبة القصوى
    contamination_tolerance DECIMAL(5,2), -- نسبة التلوث المسموحة
    
    -- شروط التسليم والاستلام
    pickup_available BOOLEAN DEFAULT FALSE,
    delivery_available BOOLEAN DEFAULT FALSE,
    transportation_cost DECIMAL(10,2) DEFAULT 0, -- تكلفة النقل التي يدفعها النظام عند الشراء من المورد

    -- معلومات الأسعار والعقود
    contract_duration_months INTEGER,
    price_validity_days INTEGER DEFAULT 30,
    price_escalation_clause TEXT, -- بند زيادة الأسعار
    seasonal_price_variation BOOLEAN DEFAULT FALSE,
    
    -- معلومات الأولوية والحالة
    priority_level INTEGER DEFAULT 1, -- 1=عادي, 2=مهم, 3=عاجل
    is_exclusive_material BOOLEAN DEFAULT FALSE, -- حصري للمورد
    is_active BOOLEAN DEFAULT TRUE,
    
    -- تواريخ مهمة
    last_purchase_date DATE,
    last_price_update DATE DEFAULT CURRENT_DATE,
    contract_start_date DATE,
    contract_end_date DATE,
    
    -- إحصائيات (يمكن تحديثها عبر محفزات أو وظائف مجدولة)
    total_purchases_amount DECIMAL(15,2) DEFAULT 0,
    total_quantity_purchased DECIMAL(15,2) DEFAULT 0,
    average_monthly_quantity DECIMAL(10,2) DEFAULT 0,
    
    notes TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    UNIQUE(supplier_id, product_id)
);

-- جدول مستندات الموردين (مع رابط الصور)

    id SERIAL PRIMARY KEY,
    supplier_id INTEGER REFERENCES suppliers(id) ON DELETE CASCADE,
    document_type_id INTEGER REFERENCES supplier_document_types(id),
    
    -- معلومات المستند
    document_name TEXT NOT NULL,
    document_number TEXT,
    description TEXT,
    
    -- معلومات الملف
    file_name TEXT,
    file_size INTEGER,
    file_type TEXT,
    file_extension TEXT,
    
    -- رابط التخزين (Storage URL)
    storage_url TEXT, -- رابط الملف في التخزين السحابي
    thumbnail_url TEXT, -- رابط الصورة المصغرة للمعاينة
    download_url TEXT, -- رابط التحميل المباشر
    
    -- معلومات إضافية للملف
    storage_provider TEXT DEFAULT 'local', -- local, aws_s3, google_cloud, etc.
    storage_path TEXT, -- المسار في نظام التخزين
    file_hash TEXT, -- hash للتحقق من سلامة الملف
    
    -- تواريخ المستند
    issue_date DATE,
    expiry_date DATE,
    
    -- حالة المستند والمراجعة
    status TEXT DEFAULT 'pending', -- pending, approved, rejected, expired, needs_renewal
    is_verified BOOLEAN DEFAULT FALSE,
    verification_method TEXT, -- manual, automatic, third_party
    
    -- معلومات المراجعة
    reviewed_by INTEGER,
    reviewed_at TIMESTAMP WITH TIME ZONE,
    review_notes TEXT,
    rejection_reason TEXT,
    
    -- تنبيهات انتهاء الصلاحية
    expiry_alert_sent BOOLEAN DEFAULT FALSE,
    expiry_alert_date TIMESTAMP WITH TIME ZONE,
    days_before_expiry_alert INTEGER DEFAULT 30,
    
    -- إعدادات الأمان والوصول
    is_confidential BOOLEAN DEFAULT FALSE,
    access_level TEXT DEFAULT 'normal', -- public, normal, restricted, confidential
    
    -- ملاحظات ومعلومات إضافية
    notes TEXT,
    tags TEXT[], -- علامات للتصنيف والبحث
    
    -- معلومات النظام
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    created_by INTEGER,
    updated_by INTEGER,
    
    UNIQUE(supplier_id, document_type_id, document_number)
);

-- جدول تقييمات الموردين
CREATE TABLE supplier_evaluations (
    id SERIAL PRIMARY KEY,
    supplier_id INTEGER REFERENCES suppliers(id) ON DELETE CASCADE,
    
    -- معايير التقييم
    quality_rating DECIMAL(2,1) CHECK (quality_rating >= 0 AND quality_rating <= 5),
    delivery_rating DECIMAL(2,1) CHECK (delivery_rating >= 0 AND delivery_rating <= 5),
    service_rating DECIMAL(2,1) CHECK (service_rating >= 0 AND service_rating <= 5),
    price_rating DECIMAL(2,1) CHECK (price_rating >= 0 AND price_rating <= 5),
    
    -- التقييم الإجمالي
    overall_rating DECIMAL(2,1) CHECK (overall_rating >= 0 AND overall_rating <= 5),
    
    -- تفاصيل التقييم
    evaluation_period_start DATE,
    evaluation_period_end DATE,
    comments TEXT,
    
    -- معلومات المقيم
    evaluated_by INTEGER NOT NULL,
    evaluation_date DATE DEFAULT CURRENT_DATE,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- جدول الجداول الزمنية للموردين (مواعيد التسليم/الاستلام/الاجتماعات)
CREATE TABLE supplier_schedules (
  id SERIAL PRIMARY KEY,
  supplier_id INTEGER REFERENCES suppliers(id) NOT NULL,
  warehouse_id INTEGER REFERENCES public.warehouses(id) NOT NULL,
  scheduled_date DATE NOT NULL,
  scheduled_time TIME WITHOUT TIME ZONE NULL,
  estimated_quantity DECIMAL(10, 2) NULL,
  purpose TEXT NULL,
  status TEXT NULL DEFAULT 'scheduled'::TEXT,
  notes TEXT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  CONSTRAINT supplier_schedules_purpose_check CHECK (
    (purpose = ANY (ARRAY['delivery'::TEXT, 'pickup'::TEXT, 'meeting'::TEXT]))
  ),
  CONSTRAINT supplier_schedules_status_check CHECK (
    (status = ANY (
        ARRAY[
          'scheduled'::TEXT,
          'completed'::TEXT,
          'cancelled'::TEXT,
          'rescheduled'::TEXT
        ]
      ))
  )
);

-- دوال مساعدة لحساب التقييم الإجمالي
CREATE OR REPLACE FUNCTION calculate_overall_rating(
    quality DECIMAL, 
    delivery DECIMAL, 
    service DECIMAL, 
    price DECIMAL
) 
RETURNS DECIMAL AS $$
BEGIN
    IF quality IS NULL OR delivery IS NULL OR service IS NULL OR price IS NULL THEN
        RETURN NULL;
    END IF;
    RETURN ROUND((quality + delivery + service + price) / 4.0, 1);
END;
$$ LANGUAGE plpgsql;

-- محفز لتحديث التقييم الإجمالي تلقائياً
CREATE OR REPLACE FUNCTION update_overall_rating()
RETURNS TRIGGER AS $$
BEGIN
    NEW.overall_rating := calculate_overall_rating(
        NEW.quality_rating, 
        NEW.delivery_rating, 
        NEW.service_rating, 
        NEW.price_rating
    );
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_update_overall_rating
    BEFORE INSERT OR UPDATE ON supplier_evaluations
    FOR EACH ROW
    EXECUTE FUNCTION update_overall_rating();

-- محفز لتحديث تاريخ التعديل
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE VIEW supplier_summary AS SELECT s.id, s.name, s.supplier_code, st.type_name_ar as supplier_type, st.type_code, s.contact_person, s.contact_phone, s.email, s.rating, s.is_active, s.is_approved, COUNT(DISTINCT sd.id) as documents_count, COUNT(DISTINCT CASE WHEN sd.status = 'approved' THEN sd.id END) as approved_documents, COUNT(DISTINCT CASE WHEN sd.expiry_date < CURRENT_DATE AND sd.status = 'approved' THEN sd.id END) as expired_documents, COUNT(DISTINCT spo.id) as products_count FROM suppliers s LEFT JOIN supplier_types st ON s.supplier_type_id = st.id LEFT JOIN supplier_documents sd ON s.id = sd.supplier_id LEFT JOIN supplier_product_offerings spo ON s.id = spo.supplier_id GROUP BY s.id, st.type_name_ar, st.type_code;

-- مشهد للمستندات منتهية الصلاحية
CREATE VIEW expired_documents AS
SELECT 
    s.name as supplier_name,
    s.supplier_code,
    dt.type_name_ar as document_type,
    sd.document_name,
    sd.expiry_date,
    CURRENT_DATE - sd.expiry_date as days_overdue
FROM supplier_documents sd
JOIN suppliers s ON sd.supplier_id = s.id
JOIN supplier_document_types dt ON sd.document_type_id = dt.id
WHERE sd.expiry_date < CURRENT_DATE
AND sd.status = 'approved';

-- مشهد إحصائيات أنواع الموردين
CREATE VIEW supplier_type_statistics AS
SELECT 
    st.type_name_ar,
    st.type_code,
    COUNT(s.id) as total_suppliers,
    COUNT(CASE WHEN s.is_active = true THEN 1 END) as active_suppliers,
    COUNT(CASE WHEN s.is_approved = true THEN 1 END) as approved_suppliers,
    AVG(s.rating) as average_rating,
    COUNT(DISTINCT spo.product_id) as unique_products
FROM supplier_types st
LEFT JOIN suppliers s ON st.id = s.supplier_type_id
LEFT JOIN supplier_product_offerings spo ON s.id = spo.supplier_id
WHERE st.is_active = true
GROUP BY st.id, st.type_name_ar, st.type_code
ORDER BY st.sort_order;

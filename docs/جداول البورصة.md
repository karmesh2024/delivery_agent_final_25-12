-- جدول عروض الأسعار من الموردين (Supplier Price Offers)
-- هذا الجدول يمثل العروض المقدمة من الموردين لأسعار المنتجات/المواد المختلفة.
-- يعتمد على جداول الموردين (suppliers) وجداول المنتجات العامة (public.categories, public.subcategories, public.waste_data_admin).
CREATE TABLE supplier_price_offers (
  id SERIAL PRIMARY KEY,
  supplier_id INTEGER REFERENCES suppliers(id) NOT NULL,
  category_id UUID REFERENCES public.categories(id) NOT NULL, -- معرف الفئة من جدول الفئات العام
  subcategory_id UUID REFERENCES public.subcategories(id) NOT NULL, -- معرف الفئة الفرعية من جدول الفئات الفرعية العام
  product_id UUID REFERENCES public.waste_data_admin(id) NOT NULL, -- معرف المنتج من جدول المنتجات العام
  proposed_price DECIMAL(10,2) NOT NULL, -- السعر المقترح من المورد
  quantity DECIMAL(10,2) NOT NULL, -- الكمية المرتبطة بهذا العرض
  unit TEXT NOT NULL, -- وحدة القياس (مثل: كجم, طن, قطعة)
  valid_from DATE NOT NULL, -- تاريخ بدء سريان العرض
  valid_to DATE NOT NULL, -- تاريخ انتهاء سريان العرض
  status TEXT CHECK (status IN ('pending', 'active', 'rejected', 'expired', 'accepted')) DEFAULT 'pending', -- حالة العرض
  notes TEXT, -- ملاحظات إضافية حول العرض
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- إنشاء الفهارس للأداء لجدول supplier_price_offers
CREATE INDEX idx_supplier_price_offers_supplier ON supplier_price_offers(supplier_id);
CREATE INDEX idx_supplier_price_offers_product ON supplier_price_offers(product_id);
CREATE INDEX idx_supplier_price_offers_status ON supplier_price_offers(status);
CREATE INDEX idx_supplier_price_offers_validity ON supplier_price_offers(valid_from, valid_to);

-- سياسات أمن مستوى الصف (RLS) لجدول supplier_price_offers
ALTER TABLE supplier_price_offers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "عروض الأسعار مرئية للمستخدمين" ON supplier_price_offers FOR SELECT USING (true);
CREATE POLICY "عروض الأسعار قابلة للإدراج من المسؤولين" ON supplier_price_offers FOR INSERT WITH CHECK (true);
CREATE POLICY "عروض الأسعار قابلة للتحديث من المسؤولين" ON supplier_price_offers FOR UPDATE USING (true);
CREATE POLICY "عروض الأسعار قابلة للحذف من المسؤولين" ON supplier_price_offers FOR DELETE USING (true);

-- محفز لتحديث تاريخ التعديل لجدول supplier_price_offers
CREATE TRIGGER update_supplier_price_offers_updated_at BEFORE UPDATE ON supplier_price_offers FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();


-- جدول البورصة
CREATE TABLE stock_exchange (
  id SERIAL PRIMARY KEY,
  category_id UUID REFERENCES categories(id) NOT NULL,
  subcategory_id UUID REFERENCES subcategories(id) NOT NULL,
  product_id UUID REFERENCES waste_data_admin(id) NOT NULL,
  region_id INTEGER REFERENCES regions(id) NOT NULL,
  base_price DECIMAL(10,2) NOT NULL,
  buy_price DECIMAL(10,2) NOT NULL,  -- سعر الشراء من العميل
  sell_price DECIMAL(10,2) NOT NULL,  -- سعر البيع للمورد
  demand_level TEXT CHECK (demand_level IN ('low', 'normal', 'high', 'critical')),
  supply_level TEXT CHECK (supply_level IN ('low', 'normal', 'high', 'over_supplied')),
  volatility_index DECIMAL(5,2) DEFAULT 0,  -- مؤشر تقلب السعر
  price_change_percentage DECIMAL(5,2) DEFAULT 0,
  auto_update_enabled BOOLEAN DEFAULT TRUE,
  last_update TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  next_update TIMESTAMP WITH TIME ZONE,
  UNIQUE(product_id, region_id)
);

-- دالة لتحديث أسعار البورصة تلقائيًا
CREATE OR REPLACE FUNCTION update_exchange_prices()
RETURNS void AS $$
DECLARE
  product_rec RECORD;
  supply_level TEXT;
  demand_level TEXT;
  new_buy_price DECIMAL(10,2);
  new_sell_price DECIMAL(10,2);
BEGIN
  -- جلب كل المنتجات في البورصة
  FOR product_rec IN SELECT * FROM stock_exchange WHERE auto_update_enabled = TRUE
  LOOP
    -- حساب مستوى العرض (المخزون)
    SELECT 
      CASE 
        WHEN SUM(wi.quantity) < 200 THEN 'low'
        WHEN SUM(wi.quantity) < 500 THEN 'normal'
        WHEN SUM(wi.quantity) < 1000 THEN 'high'
        ELSE 'over_supplied'
      END INTO supply_level
    FROM warehouse_inventory wi 
    JOIN warehouses w ON wi.warehouse_id = w.id
    WHERE wi.product_id = product_rec.product_id 
    AND w.region_id = product_rec.region_id;
    
    -- حساب مستوى الطلب (عروض الموردين)
    SELECT 
      CASE 
        WHEN COUNT(*) = 0 THEN 'low'
        WHEN COUNT(*) < 3 THEN 'normal'
        WHEN COUNT(*) < 5 THEN 'high'
        ELSE 'critical'
      END INTO demand_level
    FROM supplier_price_offers spo
    WHERE spo.product_id = product_rec.product_id 
    AND spo.status = 'active'
    AND current_date BETWEEN spo.valid_from AND spo.valid_to;

    -- حساب السعر الجديد للشراء من العملاء
    new_buy_price := product_rec.base_price;
    
    -- تأثير العرض
    IF supply_level = 'low' THEN 
      new_buy_price := new_buy_price * 1.10;
    ELSIF supply_level = 'over_supplied' THEN
      new_buy_price := new_buy_price * 0.90;
    END IF;
    
    -- تأثير الطلب
    IF demand_level = 'high' OR demand_level = 'critical' THEN
      new_buy_price := new_buy_price * 1.08;
    ELSIF demand_level = 'low' THEN
      new_buy_price := new_buy_price * 0.95;
    END IF;
    
    -- حساب سعر البيع للموردين (هامش ربح 20%)
    new_sell_price := new_buy_price * 1.2;
    
    -- تحديث جدول البورصة
    UPDATE stock_exchange
    SET 
      buy_price = new_buy_price,
      sell_price = new_sell_price,
      supply_level = supply_level,
      demand_level = demand_level,
      price_change_percentage = ((new_buy_price - base_price) / base_price) * 100,
      last_update = NOW(),
      next_update = NOW() + INTERVAL '6 hours'
    WHERE id = product_rec.id;
    
    -- إنشاء تنبيه إذا كان التغيير كبيرًا
    IF ABS((new_buy_price - product_rec.buy_price) / product_rec.buy_price) > 0.1 THEN
      INSERT INTO system_alerts (
        alert_type, severity, message, region_id, category_id, subcategory_id, product_id
      ) VALUES (
        'price_change',
        'medium',
        'تغير سعر المنتج بنسبة ' || 
        ROUND(ABS((new_buy_price - product_rec.buy_price) / product_rec.buy_price) * 100, 1) || 
        '% للمنتج: ' || (SELECT name FROM waste_data_admin WHERE id = product_rec.product_id),
        product_rec.region_id,
        product_rec.category_id,
        product_rec.subcategory_id,
        product_rec.product_id
      );
    END IF;
  END LOOP;
END;
$$ LANGUAGE plpgsql;

-- جدول تاريخ أسعار البورصة
CREATE TABLE exchange_price_history (
  id SERIAL PRIMARY KEY,
  product_id UUID REFERENCES waste_data_admin(id) NOT NULL,
  region_id INTEGER REFERENCES regions(id) NOT NULL,
  buy_price DECIMAL(10,2) NOT NULL,  -- سعر الشراء من العميل
  sell_price DECIMAL(10,2) NOT NULL,  -- سعر البيع للمورد
  demand_level TEXT,
  supply_level TEXT,
  recorded_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Trigger لتسجيل تاريخ تغيرات الأسعار
CREATE OR REPLACE FUNCTION record_price_history()
RETURNS TRIGGER AS $$
BEGIN
  IF (OLD.buy_price <> NEW.buy_price OR OLD.sell_price <> NEW.sell_price) THEN
    INSERT INTO exchange_price_history (
      product_id, region_id, buy_price, sell_price, demand_level, supply_level
    ) VALUES (
      NEW.product_id, NEW.region_id, NEW.buy_price, NEW.sell_price, NEW.demand_level, NEW.supply_level
    );
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_record_price_history
AFTER UPDATE ON stock_exchange
FOR EACH ROW
EXECUTE FUNCTION record_price_history();

    id SERIAL PRIMARY KEY,
    order_code TEXT NOT NULL UNIQUE, -- رمز أمر الشراء
    supplier_id INTEGER REFERENCES suppliers(id) NOT NULL, -- المورد الذي تم الشراء منه

    notes TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

## 🎯 نظرة عامة

هذا المستند يحدد **ما يجب تنفيذه فوراً** في الداش بورد لدعم نظام النقاط الموحد (النظام الهجين).

### المبادئ الأساسية:

1. **المحفظة المالية** (`wallet_balance`) - فلوس حقيقية قابلة للسحب
2. **نقاط المتجر** (`store_points`) - للتحفيز فقط (متجر، هدايا)
3. **فصل تام** بين النوعين
4. **تجميد السعر** عند الجلسة
5. **تسوية الجلسات** قبل السماح بالسحب

### 2. is_settled قبل السحب ⭐⭐⭐⭐⭐

**الهدف:** منع السحب النقدي إلا بعد تسوية الجلسة

**التنفيذ:**

#### A. Function: التحقق من الجلسات المعلقة

```sql
CREATE OR REPLACE FUNCTION check_unsettled_sessions(p_customer_id UUID)
RETURNS INTEGER AS $$
DECLARE
    v_count INTEGER;
BEGIN
    SELECT COUNT(*) INTO v_count
    FROM waste_collection_sessions
    WHERE customer_id = p_customer_id
      AND status = 'completed'
      AND is_settled = false;
    
    RETURN v_count;
END;
$$ LANGUAGE plpgsql;
```

#### B. Function: سحب من المحفظة (مع التحقق)

```sql
CREATE OR REPLACE FUNCTION withdraw_from_wallet(
    p_customer_id UUID,
    p_amount_egp DECIMAL(10,2)
)
RETURNS UUID AS $$
DECLARE
    v_unsettled_count INTEGER;
    v_current_wallet DECIMAL(12,2);
    v_points_per_egp INTEGER;
    v_points_required INTEGER;
    v_redemption_id UUID;
BEGIN
    -- 1. التحقق من الجلسات غير المقفلة
    v_unsettled_count := check_unsettled_sessions(p_customer_id);
    
    IF v_unsettled_count > 0 THEN
        RAISE EXCEPTION 'يوجد % جلسة غير مقفلة. لا يمكن السحب إلا بعد تسوية الجلسات', 
            v_unsettled_count;
    END IF;
    
    -- 2. التحقق من الرصيد
    SELECT wallet_balance INTO v_current_wallet
    FROM profiles
    WHERE id = p_customer_id;
    
    IF v_current_wallet < p_amount_egp THEN
        RAISE EXCEPTION 'رصيد المحفظة غير كافٍ. المتاح: % ج.م، المطلوب: % ج.م', 
            v_current_wallet, p_amount_egp;
    END IF;
    
    -- 3. جلب معامل التحويل
    SELECT points_per_egp INTO v_points_per_egp
    FROM points_configuration
    WHERE is_active = true
    ORDER BY created_at DESC
    LIMIT 1;
    
    v_points_per_egp := COALESCE(v_points_per_egp, 100);
    v_points_required := (p_amount_egp * v_points_per_egp)::INTEGER;
    
    -- 4. إنشاء سجل الاستبدال
    v_redemption_id := gen_random_uuid();
    
    INSERT INTO points_redemptions (
        id,
        customer_id,
        points_redeemed,
        amount_egp,
        redemption_type,
        status
    ) VALUES (
        v_redemption_id,
        p_customer_id,
        v_points_required,
        p_amount_egp,
        'cash',
        'pending'
    );
    
    -- 5. خصم من المحفظة
    UPDATE profiles
    SET 
        wallet_balance = wallet_balance - p_amount_egp,
        updated_at = NOW()
    WHERE id = p_customer_id;
    
    -- 6. تسجيل في Audit Log
    INSERT INTO points_transactions (
        profile_id,
        type,
        amount,
        before_balance,
        after_balance,
        source,
        reference_id,
        description
    )
    SELECT 
        p_customer_id,
        'wallet',
        -v_points_required,
        p.wallet_balance,
        p.wallet_balance - p_amount_egp,
        'points_redemption',
        v_redemption_id,
        'سحب نقدي من المحفظة: ' || p_amount_egp || ' ج.م'
    FROM profiles p
    WHERE p.id = p_customer_id;
    
    RETURN v_redemption_id;
END;
$$ LANGUAGE plpgsql;
```

**النتيجة:**
- ✅ لا يمكن السحب إلا بعد `is_settled = true`
- ✅ رسالة خطأ واضحة للعميل

#### B. Trigger: تسجيل تلقائي عند تحديث wallet_balance

```sql
CREATE OR REPLACE FUNCTION log_wallet_transaction()
RETURNS TRIGGER AS $$
BEGIN
    -- تسجيل فقط إذا تغير wallet_balance
    IF OLD.wallet_balance IS DISTINCT FROM NEW.wallet_balance THEN
        INSERT INTO points_transactions (
            profile_id,
            type,
            amount,
            before_balance,
            after_balance,
            source,
            description
        ) VALUES (
            NEW.id,
            'wallet',
            ((NEW.wallet_balance - OLD.wallet_balance) * 100)::INTEGER,
            (OLD.wallet_balance * 100)::INTEGER,
            (NEW.wallet_balance * 100)::INTEGER,
            'system',
            'تحديث تلقائي للمحفظة'
        );
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trigger_log_wallet_transaction ON profiles;
CREATE TRIGGER trigger_log_wallet_transaction
AFTER UPDATE ON profiles
FOR EACH ROW
WHEN (OLD.wallet_balance IS DISTINCT FROM NEW.wallet_balance)
EXECUTE FUNCTION log_wallet_transaction();
```

### 4. Freeze unit_price ⭐⭐⭐⭐⭐

**الهدف:** تجميد سعر البورصة عند الجلسة

**التنفيذ:**

#### A. Function: حساب عنصر التجميع (مع تجميد السعر)

```sql
CREATE OR REPLACE FUNCTION calculate_collection_item()
RETURNS TRIGGER AS $$
DECLARE
    v_product RECORD;
    v_buy_price DECIMAL(10,2);
    v_total_weight DECIMAL(10,3);
    v_item_value DECIMAL(10,2);
BEGIN
    -- 1. جلب بيانات المنتج
    SELECT 
        wda.weight,
        wda.subcategory_id,
        wda.counting_method
    INTO v_product
    FROM waste_data_admin wda
    WHERE wda.id = NEW.waste_data_id;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION 'المنتج غير موجود: %', NEW.waste_data_id;
    END IF;
    
    -- 2. حساب الوزن
    IF v_product.counting_method = 'pieces' THEN
        v_total_weight := (v_product.weight / 1000.0) * NEW.quantity;
    ELSIF v_product.counting_method = 'weight' THEN
        v_total_weight := NEW.actual_weight;
    END IF;
    
    NEW.total_weight := v_total_weight;
    
    -- 3. ⭐ تجميد السعر (فقط عند INSERT)
    IF TG_OP = 'INSERT' THEN
        SELECT buy_price INTO v_buy_price
        FROM stock_exchange
        WHERE sub_category_id = v_product.subcategory_id
        ORDER BY last_update DESC
        LIMIT 1;
        
        v_buy_price := COALESCE(v_buy_price, 0);
        NEW.unit_price := v_buy_price; -- ⭐ تجميد السعر
    ELSE
        -- في UPDATE: استخدام السعر المجمد الموجود
        v_buy_price := COALESCE(NEW.unit_price, OLD.unit_price, 0);
    END IF;
    
    -- 4. حساب القيمة المالية
    v_item_value := v_total_weight * v_buy_price;
    NEW.total_price := v_item_value;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trigger_calculate_collection_item ON waste_collection_items;
CREATE TRIGGER trigger_calculate_collection_item
BEFORE INSERT OR UPDATE ON waste_collection_items
FOR EACH ROW
EXECUTE FUNCTION calculate_collection_item();
```

**النتيجة:**
- ✅ السعر مجمد عند إنشاء العنصر
- ✅ لا يتأثر بالتحديثات اللاحقة

```typescript
async function processWithdrawal(redemptionId: string) {
  // 1. التحقق من البيانات
  const redemption = await getRedemption(redemptionId);
  
  // 2. التحقق من الجلسات المعلقة
  const unsettledCount = await supabase.rpc('check_unsettled_sessions', {
    p_customer_id: redemption.customer_id
  });
  
  if (unsettledCount.data > 0) {
    throw new Error('يوجد جلسات غير مقفلة');
  }
  
  // 3. التحويل البنكي
  const bankTransfer = await bankAPI.transfer({
    to: redemption.customer_bank_account,
    amount: redemption.amount_egp,
    reference: `RED-${redemptionId}`
  });
  
  // 4. تحديث الحالة
  await supabase
    .from('points_redemptions')
    .update({
      status: 'completed',
      reference_number: bankTransfer.id
    })
    .eq('id', redemptionId);
  
  // 5. خصم من المحفظة (يتم تلقائياً عبر Function)
  await supabase.rpc('withdraw_from_wallet', {
    p_customer_id: redemption.customer_id,
    p_amount_egp: redemption.amount_egp
  });
}
```

### 2. calculate_session_points()

**الوظيفة:** حساب النقاط الأساسية والبونص عند إتمام الجلسة.  
اعتباراً من migration `hybrid_points_system_integrate_points_per_kg.sql`، صار **نقاط كل كجم** (`points_per_kg`) من إعدادات النقاط (`points_configurations`) جزءاً من البونص: تُحسب كنقاط هدية من وزن كل عنصر وتُضاف إلى `bonus_points`. التفاصيل والأمثلة في `docs/تسعير_النقاط_لكل_كجم_في_النظام_الجديد.md`.

```sql
CREATE OR REPLACE FUNCTION calculate_session_points()
RETURNS TRIGGER AS $$
DECLARE
    v_total_value DECIMAL(12,2);
    v_points_per_egp INTEGER;
    v_base_points INTEGER;
    v_bonus_points INTEGER;
BEGIN
    -- فقط عند تغيير الحالة إلى completed
    IF NEW.status = 'completed' AND 
       (OLD.status IS NULL OR OLD.status != 'completed') THEN
        
        -- 1. حساب القيمة المالية الإجمالية
        SELECT COALESCE(SUM(total_price), 0) INTO v_total_value
        FROM waste_collection_items
        WHERE session_id = NEW.id;
        
        NEW.total_value := v_total_value;
        
        -- 2. جلب معامل التحويل
        SELECT points_per_egp INTO v_points_per_egp
        FROM points_configuration
        WHERE is_active = true
        ORDER BY created_at DESC
        LIMIT 1;
        
        v_points_per_egp := COALESCE(v_points_per_egp, 100);
        
        -- 3. حساب النقاط الأساسية
        v_base_points := FLOOR(v_total_value * v_points_per_egp);
        NEW.base_points := v_base_points;
        
        -- 4. حساب نقاط المتجر (البونص) - للبداية: 0
        v_bonus_points := 0; -- ⚠️ للبداية: لا بونص
        NEW.bonus_points := v_bonus_points;
        NEW.total_points := v_base_points;
        
        -- 5. منح نقاط المتجر فوراً (إذا كان هناك بونص)
        IF v_bonus_points > 0 THEN
            UPDATE profiles
            SET 
                store_points = COALESCE(store_points, 0) + v_bonus_points,
                updated_at = NOW()
            WHERE id = NEW.customer_id;
        END IF;
        
        -- 6. تسجيل في Audit Log (للمحفظة - معلق)
        INSERT INTO points_transactions (
            profile_id,
            type,
            amount,
            before_balance,
            after_balance,
            source,
            reference_id,
            description
        )
        SELECT 
            NEW.customer_id,
            'wallet',
            v_base_points,
            (COALESCE(p.wallet_balance, 0) * 100)::INTEGER,
            (COALESCE(p.wallet_balance, 0) * 100)::INTEGER, -- لا يزال معلق
            'waste_collection_session',
            NEW.id,
            'نقاط محفظة من جلسة تجميع (معلق حتى التسوية)'
        FROM profiles p
        WHERE p.id = NEW.customer_id;
        
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```

### 3. settle_collection_session()

**الوظيفة:** تسوية الجلسة - جعل نقاط المحفظة متاحة

```sql
CREATE OR REPLACE FUNCTION settle_collection_session(p_session_id UUID)
RETURNS BOOLEAN AS $$
DECLARE
    v_session RECORD;
    v_base_points INTEGER;
BEGIN
    -- 1. جلب بيانات الجلسة
    SELECT 
        id,
        customer_id,
        base_points,
        is_settled
    INTO v_session
    FROM waste_collection_sessions
    WHERE id = p_session_id;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION 'الجلسة غير موجودة: %', p_session_id;
    END IF;
    
    IF v_session.is_settled = true THEN
        RAISE EXCEPTION 'الجلسة مقفلة بالفعل';
    END IF;
    
    IF v_session.status != 'completed' THEN
        RAISE EXCEPTION 'الجلسة لم تكتمل بعد';
    END IF;
    
    v_base_points := COALESCE(v_session.base_points, 0);
    
    -- 2. تحديث المحفظة المالية (متاح للسحب الآن)
    UPDATE profiles
    SET 
        wallet_balance = COALESCE(wallet_balance, 0) + (v_base_points::DECIMAL / 100),
        updated_at = NOW()
    WHERE id = v_session.customer_id;
    
    -- 3. تحديث حالة الجلسة
    UPDATE waste_collection_sessions
    SET 
        is_settled = true,
        updated_at = NOW()
    WHERE id = p_session_id;
    
    -- 4. تحديث Audit Log
    UPDATE points_transactions
    SET 
        after_balance = (
            SELECT (wallet_balance * 100)::INTEGER
            FROM profiles 
            WHERE id = v_session.customer_id
        ),
        description = description || ' - تم التسوية'
    WHERE reference_id = p_session_id
      AND type = 'wallet'
      AND source = 'waste_collection_session';
    
    RETURN true;
END;
$$ LANGUAGE plpgsql;
```

### 6. use_store_points()

**الوظيفة:** استخدام نقاط المتجر

```sql
CREATE OR REPLACE FUNCTION use_store_points(
    p_customer_id UUID,
    p_points INTEGER,
    p_redemption_type VARCHAR DEFAULT 'product'
)
RETURNS UUID AS $$
DECLARE
    v_current_store_points INTEGER;
    v_redemption_id UUID;
BEGIN
    -- 1. التحقق من رصيد نقاط المتجر
    SELECT store_points INTO v_current_store_points
    FROM profiles
    WHERE id = p_customer_id;
    
    IF v_current_store_points < p_points THEN
        RAISE EXCEPTION 'رصيد نقاط المتجر غير كافٍ. المتاح: %, المطلوب: %', 
            v_current_store_points, p_points;
    END IF;
    
    -- 2. إنشاء سجل الاستبدال
    v_redemption_id := gen_random_uuid();
    
    INSERT INTO points_redemptions (
        id,
        customer_id,
        points_redeemed,
        amount_egp,
        redemption_type,
        status
    ) VALUES (
        v_redemption_id,
        p_customer_id,
        p_points,
        0, -- نقاط المتجر لا قيمة نقدية
        p_redemption_type,
        'pending'
    );
    
    -- 3. خصم من نقاط المتجر
    UPDATE profiles
    SET 
        store_points = store_points - p_points,
        updated_at = NOW()
    WHERE id = p_customer_id;
    
    -- 4. تسجيل في Audit Log
    INSERT INTO points_transactions (
        profile_id,
        type,
        amount,
        before_balance,
        after_balance,
        source,
        reference_id,
        description
    )
    SELECT 
        p_customer_id,
        'store',
        -p_points,
        p.store_points,
        p.store_points - p_points,
        'points_redemption',
        v_redemption_id,
        'استخدام نقاط المتجر: ' || p_redemption_type
    FROM profiles p
    WHERE p.id = p_customer_id;
    
    RETURN v_redemption_id;
END;
$$ LANGUAGE plpgsql;
```

```typescript
  const redemption = await getRedemption(redemptionId);
  
  if (action === 'approve') {
    // 1. التحقق من الجلسات المعلقة
    const { data: unsettledCount } = await supabase.rpc('check_unsettled_sessions', {
```

# تقييم الوضع الحالي وخطوات تنفيذ نظام الصلاحيات

## تقييم ما تم إنجازه حتى الآن

بعد مراجعة الملفات الحالية للمشروع، تبين أن هناك تقدم في إعداد نظام الصلاحيات، خاصة على مستوى قاعدة البيانات:

### 1. ما تم إنجازه في قاعدة البيانات

1. **هيكل الجداول**:
   - تم إنشاء جدول `roles` بالأعمدة المطلوبة (id, name, code, description, permissions)
   - تم إضافة عمود `permissions` من نوع jsonb إلى جدول المسؤولين `admins`
   - تم إضافة عمود `role_id` في جدول `admins` للربط مع جدول `roles`

2. **إعداد البيانات الأساسية**:
   - تم إنشاء دور "مدير النظام" (super_admin) مع صلاحيات كاملة
   - تم تعيين مستخدم فعلي (despitalia1@gmail.com) كمدير للنظام مع صلاحيات كاملة

3. **دوال التحقق من الصلاحيات**:
   - تم تنفيذ دالة `check_admin_permission` للتحقق من صلاحيات المسؤول
   - تم تنفيذ دالة محسنة `check_admin_permission_enhanced` تدعم هيكل تسلسل المديرين

### 2. ما لم يتم إنجازه بعد

1. **دوال إضافية في قاعدة البيانات**:
   - لم يتم تنفيذ دالة `check_permission_with_scope` للتحقق من الصلاحيات في نطاق محدد
   - ربما ستكون هناك حاجة لدوال إضافية لإدارة الصلاحيات

2. **واجهة برمجة التطبيق (API)**:
   - لم يتم تنفيذ وظائف في TypeScript للتفاعل مع نظام الصلاحيات في قاعدة البيانات
   - لم يتم تحديث ملفات الأنواع (types) لدعم الصلاحيات والأدوار

3. **تكامل Redux**:
   - لم يتم تنفيذ تكامل نظام الصلاحيات مع Redux
   - لا توجد شرائح (slices) مخصصة لإدارة الصلاحيات والأدوار

4. **واجهة المستخدم**:
   - لم يتم تنفيذ مكونات لعرض وإدارة الصلاحيات
   - لم يتم تنفيذ الحماية بالصلاحيات في واجهة المستخدم والصفحات

1. **تنفيذ دالة التحقق من الصلاحيات مع النطاق**:
   ```sql
   -- تنفيذ في ملف جديد src/domains/admins/data/check-permission-scope.sql
   CREATE OR REPLACE FUNCTION check_permission_with_scope(
     p_admin_id uuid,
     p_permission_code text,
     p_scope_type text,
     p_scope_value jsonb
   )
   RETURNS boolean AS $$
   DECLARE
     has_permission boolean;
     has_scope_permission boolean;
     permissions jsonb;
     role_permissions jsonb;
     role_id uuid;
   BEGIN
     -- 1. جلب معلومات المسؤول
     SELECT a.permissions, a.role_id INTO permissions, role_id
     FROM admins a WHERE a.id = p_admin_id;

     IF permissions IS NULL AND role_id IS NULL THEN
       RETURN false;
     END IF;

     -- 2. التحقق من الصلاحية العامة
     SELECT check_admin_permission_enhanced(p_admin_id, p_permission_code) INTO has_permission;
     
     IF NOT has_permission THEN
       RETURN false;
     END IF;

     -- 3. التحقق من الصلاحية بالنطاق
     -- يعتمد هذا على هيكل تخزين نطاقات الصلاحيات، في هذا المثال نفترض أنه يتم تخزينها في حقل scoped_permissions
     
     -- في البداية، تحقق من الصلاحيات الخاصة بالمسؤول
     IF permissions IS NOT NULL AND 
        permissions ? 'scoped_permissions' AND
        permissions->'scoped_permissions' ? p_scope_type AND
        (permissions->'scoped_permissions'->p_scope_type)::jsonb ? p_permission_code THEN
       
       -- تحقق من قيمة النطاق
       DECLARE
         scope_values jsonb;
       BEGIN
         scope_values := (permissions->'scoped_permissions'->p_scope_type->p_permission_code);
         
         -- إذا كانت قيمة النطاق تحتوي على '*' فهذا يعني أن لديه صلاحية على جميع النطاقات
         IF scope_values ? '*' OR scope_values @> p_scope_value THEN
           RETURN true;
         END IF;
       END;
     END IF;

     -- تحقق من صلاحيات الدور إذا لم تكن الصلاحية موجودة في صلاحيات المسؤول المخصصة
     IF role_id IS NOT NULL THEN
       SELECT permissions INTO role_permissions
       FROM roles r WHERE r.id = role_id;
       
       IF role_permissions IS NOT NULL AND 
          role_permissions ? 'scoped_permissions' AND
          role_permissions->'scoped_permissions' ? p_scope_type AND
          (role_permissions->'scoped_permissions'->p_scope_type)::jsonb ? p_permission_code THEN
         
         DECLARE
           scope_values jsonb;
         BEGIN
           scope_values := (role_permissions->'scoped_permissions'->p_scope_type->p_permission_code);
           
           IF scope_values ? '*' OR scope_values @> p_scope_value THEN
             RETURN true;
           END IF;
         END;
       END IF;
     END IF;

     -- إذا وصلنا إلى هنا، فلا توجد صلاحية بالنطاق المطلوب
     RETURN false;
   END;
   $$ LANGUAGE plpgsql;
   ```

   ```typescript
   /**
    * التحقق من امتلاك المسؤول للصلاحية
    */
   export async function hasPermission(admin: Admin | null, permissionCode: string): Promise<boolean> {
     if (!admin) return false;
     
     // تحقق محلي (تحقق أولي)
     if (admin.permissions && admin.permissions[permissionCode] === true) {
       return true;
     }
     
     try {
       // استعلام من قاعدة البيانات
       const { data, error } = await supabase.rpc('check_admin_permission_enhanced', {
         admin_id: admin.id,
         permission_code: permissionCode
       });
       
       if (error) {
         console.error('خطأ في التحقق من الصلاحية:', error);
         return false;
       }
       
       return !!data;
     } catch (error) {
       console.error('خطأ غير متوقع:', error);
       return false;
     }
   }

   /**
    * فحص صلاحية مع نطاق بيانات
    */
   export async function hasPermissionWithScope(
     admin: Admin | null, 
     permissionCode: string,
     scopeType: string,
     scopeValue: any
   ): Promise<boolean> {
     if (!admin) return false;
     
     try {
       const { data, error } = await supabase.rpc('check_permission_with_scope', {
         p_admin_id: admin.id,
         p_permission_code: permissionCode,
         p_scope_type: scopeType,
         p_scope_value: scopeValue
       });
       
       if (error) {
         console.error('خطأ في التحقق من الصلاحية مع النطاق:', error);
         return false;
       }
       
       return !!data;
     } catch (error) {
       console.error('خطأ غير متوقع:', error);
       return false;
     }
   }

   /**
    * التحقق المحلي من الصلاحيات (بدون استعلام قاعدة البيانات)
    */
   ```

   ```typescript
   import { supabase } from '@/lib/supabase';
   import { AdminPermissions } from '../types/permissions';

   export const permissionsApi = {
     /**
      * جلب الصلاحيات المتاحة
      */
     getAvailablePermissions: async () => {
       try {
         const { data, error } = await supabase
           .from('permissions')
           .select('*')
           .order('code');
         
         if (error) throw error;
         return { success: true, permissions: data };
       } catch (error) {
         console.error('خطأ في جلب الصلاحيات المتاحة:', error);
         return { success: false, message: 'فشل جلب الصلاحيات', permissions: [] };
       }
     },

     /**
      * إدارة صلاحيات المسؤول
      */
     managePermissions: async (adminId: string, permissions: AdminPermissions) => {
       try {
         // تحديث صلاحيات المسؤول
         const { error } = await supabase
           .from('admins')
           .update({ permissions })
           .eq('id', adminId);
         
         if (error) throw error;
         return { success: true };
       } catch (error) {
         console.error('خطأ في تحديث صلاحيات المسؤول:', error);
         return { success: false, message: 'فشل تحديث الصلاحيات' };
       }
     },

     /**
      * جلب الأدوار المتاحة
      */
     getRoles: async () => {
       try {
         const { data, error } = await supabase
           .from('roles')
           .select('*')
           .order('level', { ascending: false });
         
         if (error) throw error;
         return { success: true, roles: data };
       } catch (error) {
         console.error('خطأ في جلب الأدوار:', error);
         return { success: false, message: 'فشل جلب الأدوار', roles: [] };
       }
     },

     /**
      * تعيين دور للمسؤول
      */
     setAdminRole: async (adminId: string, roleId: string) => {
       try {
         const { error } = await supabase
           .from('admins')
           .update({ role_id: roleId })
           .eq('id', adminId);
         
         if (error) throw error;
         return { success: true };
       } catch (error) {
         console.error('خطأ في تعيين دور المسؤول:', error);
         return { success: false, message: 'فشل تعيين دور المسؤول' };
       }
     }
   };
   ```

   ```typescript
         const response = await permissionsApi.getAvailablePermissions();
         if (!response.success) {
           return rejectWithValue(response.message || 'فشل جلب الصلاحيات');
         }
         return response.permissions;

         return rejectWithValue(error instanceof Error ? error.message : 'خطأ غير معروف');
       }
     }
   );

   export const fetchRoles = createAsyncThunk(
     'permissions/fetchRoles',
     async (_, { rejectWithValue }) => {
       try {
         const response = await permissionsApi.getRoles();
         if (!response.success) {
           return rejectWithValue(response.message || 'فشل جلب الأدوار');
         }
         return response.roles;
       } catch (error) {
         return rejectWithValue(error instanceof Error ? error.message : 'خطأ غير معروف');
       }
     }
   );

   export const updateAdminPermissions = createAsyncThunk(
     'permissions/updateAdminPermissions',
     async ({ adminId, permissions }: { adminId: string, permissions: Record<string, boolean> }, { rejectWithValue }) => {
       try {
         const response = await permissionsApi.managePermissions(adminId, permissions);
         if (!response.success) {
           return rejectWithValue(response.message || 'فشل تحديث الصلاحيات');
         }
         return { adminId, permissions };
       } catch (error) {
         return rejectWithValue(error instanceof Error ? error.message : 'خطأ غير معروف');
       }
     }
   );

   // تعريف الشريحة
   const permissionsSlice = createSlice({
     name: 'permissions',
     initialState,
     reducers: {
       clearPermissionsError: (state) => {
         state.error = null;
       }
     },
     extraReducers: (builder) => {
       builder
         // fetchAvailablePermissions
         .addCase(fetchAvailablePermissions.pending, (state) => {
           state.loading = true;
           state.error = null;
         })
         .addCase(fetchAvailablePermissions.fulfilled, (state, action) => {
           state.loading = false;
           state.availablePermissions = action.payload;
         })
         .addCase(fetchAvailablePermissions.rejected, (state, action) => {
           state.loading = false;
           state.error = action.payload as string;
         })

         .addCase(fetchRoles.pending, (state) => {
           state.loading = true;
           state.error = null;
         })
         .addCase(fetchRoles.fulfilled, (state, action) => {
           state.loading = false;
           state.roles = action.payload;
         })
         .addCase(fetchRoles.rejected, (state, action) => {
           state.loading = false;
           state.error = action.payload as string;
         });
     }
   });

   export const { clearPermissionsError } = permissionsSlice.actions;
   export default permissionsSlice.reducer;
   ```

   ```tsx
   import { useAppSelector } from '@/store/hooks';
   import { hasPermissionLocal, hasPermission } from '../utils/permissions';

   interface PermissionGuardProps {
     permissionCode: string;
     children: ReactNode;
     fallback?: ReactNode;
     serverCheck?: boolean;
   }

   export function PermissionGuard({ 
     permissionCode, 
     children, 
     fallback = null, 
     serverCheck = false 
   }: PermissionGuardProps) {
     const { currentAdmin } = useAppSelector(state => state.auth);
     const [hasAccess, setHasAccess] = useState(false);
     const [loading, setLoading] = useState(serverCheck);
     
     useEffect(() => {
       // التحقق المحلي السريع أولاً
       const localCheck = hasPermissionLocal(currentAdmin, permissionCode);
       
       if (localCheck || !serverCheck) {
         setHasAccess(localCheck);
         setLoading(false);
         return;
       }
       
       // التحقق من الخادم إذا تم طلبه
       const checkServerPermission = async () => {
         try {
           const result = await hasPermission(currentAdmin, permissionCode);
           setHasAccess(result);
         } catch (error) {
           console.error('خطأ في التحقق من الصلاحية:', error);
           setHasAccess(false);
         } finally {
           setLoading(false);
         }
       };
       
       checkServerPermission();
     }, [currentAdmin, permissionCode, serverCheck]);
     
     if (loading) {
       return null; // أو مؤشر تحميل
     }
     
     return hasAccess ? <>{children}</> : <>{fallback}</>;
   }
   ```

   ```tsx
       });
     };
     
     const handleSave = async () => {
       setSuccessMessage('');
       try {
         await dispatch(updateAdminPermissions({ adminId, permissions })).unwrap();
         setSuccessMessage('تم حفظ الصلاحيات بنجاح');
       } catch (err) {
         // تم معالجة الخطأ في الشريحة
       }
     };
     
     if (loading) {
       return <div>جاري التحميل...</div>;
     }
     
     return (
       <div className="space-y-4">
         {error && <Alert variant="destructive">{error}</Alert>}
         {successMessage && <Alert variant="success">{successMessage}</Alert>}
         
         {Object.entries(permissionsByGroup).map(([group, perms]) => (
           <Card key={group}>
             <CardHeader>
               <CardTitle>{group}</CardTitle>
               <CardDescription>صلاحيات {group}</CardDescription>
             </CardHeader>
             <CardContent>
               <div className="space-y-2">
                 {perms.map(permission => (
                   <div key={permission.code} className="flex items-center justify-between py-2">
                     <div>
                       <div className="font-medium">{permission.name}</div>
                       {permission.description && (
                         <div className="text-sm text-gray-500">{permission.description}</div>
                       )}
                     </div>
                     <Switch
                       checked={!!permissions[permission.code]}
                       onCheckedChange={(checked) => handlePermissionChange(permission.code, checked)}
                     />
                   </div>
                 ))}
               </div>
             </CardContent>
           </Card>
         ))}
         
         <div className="flex justify-end">
           <Button onClick={handleSave} disabled={loading}>
             حفظ الصلاحيات
           </Button>
         </div>
       </div>
     );
   }
   ```

   ```typescript
   import { useState, useEffect } from 'react';
   import { useAppSelector } from '@/store/hooks';
   import { hasPermission, hasPermissionLocal } from '../utils/permissions';

   export function usePermission(permissionCode: string, serverCheck = false) {
     const { currentAdmin } = useAppSelector(state => state.auth);
     const [hasAccess, setHasAccess] = useState(false);
     const [loading, setLoading] = useState(serverCheck);
     
     useEffect(() => {
       // التحقق المحلي السريع أولاً
       const localCheck = hasPermissionLocal(currentAdmin, permissionCode);
       
       if (localCheck || !serverCheck) {
         setHasAccess(localCheck);
         setLoading(false);
         return;
       }
       
       // التحقق من الخادم إذا تم طلبه
       const checkServerPermission = async () => {
         try {
           const result = await hasPermission(currentAdmin, permissionCode);
           setHasAccess(result);
         } catch (error) {
           console.error('خطأ في التحقق من الصلاحية:', error);
           setHasAccess(false);
         } finally {
           setLoading(false);
         }
       };
       
       checkServerPermission();
     }, [currentAdmin, permissionCode, serverCheck]);
     
     return { hasAccess, loading };
   }
   ```

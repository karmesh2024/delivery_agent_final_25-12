import { supabase } from '@/lib/supabase';
import { Admin } from '../types';

/**
 * التحقق من امتلاك المسؤول للصلاحية
 */
export async function hasPermission(admin: Admin, permissionCode: string): Promise<boolean> {
  try {
    // استخدام الدالة المحسنة
    const { data, error } = await supabase.rpc('check_admin_permission_enhanced', {
      admin_id: admin.id,
      permission_code: permissionCode
    });
    
    if (error) {
      console.error('خطأ في التحقق من الصلاحية:', error);
      return false;
    }
    
    return !!data;
  } catch (error) {
    console.error('خطأ غير متوقع:', error);
    return false;
  }
}

/**
 * فحص صلاحية مع نطاق بيانات
 */
export async function hasPermissionWithScope(
  admin: Admin, 
  permissionCode: string,
  scopeType: string,
  scopeValue: any
): Promise<boolean> {
  try {
    const { data, error } = await supabase.rpc('check_permission_with_scope', {
      p_admin_id: admin.id,
      p_permission_code: permissionCode,
      p_scope_type: scopeType,
      p_scope_value: scopeValue
    });
    
    if (error) {
      console.error('خطأ في التحقق من الصلاحية مع النطاق:', error);
      return false;
    }
    
    return !!data;
  } catch (error) {
    console.error('خطأ غير متوقع:', error);
    return false;
  }
}

### 1.3 تحديث واجهة برمجة المسؤولين

تحديث ملف `src/domains/admins/api/adminsApi.ts` لدعم وظائف إدارة الصلاحيات:

```typescript
// إضافة الدوال التالية إلى كائن adminsApi

/**
 * إدارة صلاحيات المسؤول
 */
managePermissions: async (adminId: string, permissions: AdminPermissions): Promise<boolean> => {
  try {
    const { error } = await supabase.rpc('add_admin_permissions', {
      p_admin_id: adminId,
      p_permissions: permissions
    });
    
    if (error) throw error;
    return true;
  } catch (error) {
    console.error('خطأ في إدارة صلاحيات المسؤول:', error);
    return false;
  }
},

/**
 * جلب الصلاحيات المتاحة
 */
getAvailablePermissions: async (): Promise<any[]> => {
  try {
    const { data, error } = await supabase
      .from('permissions')
      .select(`
        id,
        code,
        name,
        description,
        resource:resources(id, code, name),
        action:actions(id, code, name)
      `)
      .order('code');
    
    if (error) throw error;
    return data;
  } catch (error) {
    console.error('خطأ في جلب الصلاحيات المتاحة:', error);
    return [];
  }
}
```

import { adminsApi } from '../api/adminsApi';

export const fetchAvailablePermissions = createAsyncThunk(
  'admins/fetchAvailablePermissions',
  async (_, { rejectWithValue }) => {
    try {
      const permissions = await adminsApi.getAvailablePermissions();
      return permissions;
    } catch (error) {
      return rejectWithValue(error instanceof Error ? error.message : 'حدث خطأ أثناء جلب الصلاحيات');
    }
  }
);

export const updateAdminPermissions = createAsyncThunk(
  'admins/updateAdminPermissions',
  async ({ adminId, permissions }: { adminId: string, permissions: AdminPermissions }, { rejectWithValue }) => {
    try {
      const success = await adminsApi.managePermissions(adminId, permissions);
      if (!success) {
        throw new Error('فشل تحديث الصلاحيات');
      }
      return { adminId, permissions };
    } catch (error) {
      return rejectWithValue(error instanceof Error ? error.message : 'حدث خطأ أثناء تحديث الصلاحيات');
    }
  }
);

// إضافة reducers للإجراءات الجديدة
const adminsSlice = createSlice({
  name: 'admins',
  initialState,
  reducers: {
    // الموجودة...
  },
  extraReducers: (builder) => {
    // إضافة معالجات للإجراءات الجديدة
    builder
      .addCase(fetchAvailablePermissions.pending, (state) => {
        state.permissions.loading = true;

      .addCase(fetchAvailablePermissions.fulfilled, (state, action) => {
        state.permissions.loading = false;

        state.permissions.loading = false;

## 3. إنشاء مكونات واجهة المستخدم

### 3.1 إنشاء مكون إدارة الصلاحيات

إنشاء ملف `src/domains/admins/components/PermissionManager.tsx`:

```tsx
import { useEffect, useState } from 'react';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { fetchAvailablePermissions, updateAdminPermissions } from '../store/adminsSlice';
import { AdminPermissions } from '../types';
import { Card, CardContent, CardHeader, CardTitle } from '@/shared/components/ui/card';
import { Switch } from '@/shared/components/ui/switch';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/shared/components/ui/tabs';
import { Button } from '@/shared/components/ui/button';

interface PermissionManagerProps {
  adminId: string;
  initialPermissions: AdminPermissions;
}

export function PermissionManager({ adminId, initialPermissions }: PermissionManagerProps) {
  const dispatch = useAppDispatch();
  const { availablePermissions, loading } = useAppSelector(state => state.admins.permissions);
  const [permissions, setPermissions] = useState<AdminPermissions>(initialPermissions || {});
  const [activeTab, setActiveTab] = useState<string>('dashboard');
  
  useEffect(() => {
    dispatch(fetchAvailablePermissions());
  }, [dispatch]);
  
  // تجميع الصلاحيات حسب المورد
  const permissionsByResource = availablePermissions.reduce((acc, permission) => {
    const resourceCode = permission.resource.code;
    if (!acc[resourceCode]) {
      acc[resourceCode] = [];
    }
    acc[resourceCode].push(permission);
    return acc;
  }, {} as Record<string, any[]>);
  
  const handleTogglePermission = (permissionCode: string, checked: boolean) => {
    setPermissions({
      ...permissions,
      [permissionCode]: checked
    });
  };
  
  const handleSavePermissions = () => {
    dispatch(updateAdminPermissions({ adminId, permissions }));
  };
  
  if (loading) {
    return <div>جاري التحميل...</div>;
  }
  
  return (
    <Card>
      <CardHeader>
        <CardTitle>إدارة صلاحيات المسؤول</CardTitle>
      </CardHeader>
      <CardContent>
        <Tabs value={activeTab} onValueChange={setActiveTab}>
          <TabsList className="mb-4">
            {Object.keys(permissionsByResource).map(resourceCode => (
              <TabsTrigger key={resourceCode} value={resourceCode}>
                {permissionsByResource[resourceCode][0].resource.name}
              </TabsTrigger>
            ))}
          </TabsList>
          
          {Object.keys(permissionsByResource).map(resourceCode => (
            <TabsContent key={resourceCode} value={resourceCode}>
              <div className="space-y-4">
                {permissionsByResource[resourceCode].map(permission => (
                  <div key={permission.id} className="flex items-center justify-between">
                    <div>
                      <div className="font-medium">{permission.name}</div>
                      <div className="text-sm text-gray-500">{permission.description}</div>
                    </div>
                    <Switch
                      checked={!!permissions[permission.code]}
                      onCheckedChange={(checked) => handleTogglePermission(permission.code, checked)}
                    />
                  </div>
                ))}
              </div>
            </TabsContent>
          ))}
        </Tabs>
        
        <div className="mt-6 flex justify-end">
          <Button onClick={handleSavePermissions}>حفظ الصلاحيات</Button>
        </div>
      </CardContent>
    </Card>
  );
}
```

## 5. تنفيذ حارس الطرق (Route Guard)

### 5.1 إنشاء حارس للطرق المحمية

إنشاء ملف `src/shared/components/RouteGuard.tsx`:

```tsx
import { useEffect, useState, ReactNode } from 'react';
import { useRouter } from 'next/router';
import { useAppSelector, useAppDispatch } from '@/store/hooks';
import { checkAuth } from '@/domains/admins/store/authSlice';
import { hasPermission } from '@/domains/admins/utils/permissions';

interface RouteGuardProps {
  children: ReactNode;
  requiredPermission?: string;
}

export function RouteGuard({ children, requiredPermission }: RouteGuardProps) {
  const router = useRouter();
  const dispatch = useAppDispatch();
  const { currentAdmin, isAuthenticated, loading } = useAppSelector(state => state.auth);
  const [authorized, setAuthorized] = useState(false);
  
  useEffect(() => {
    dispatch(checkAuth());
  }, [dispatch]);
  
  useEffect(() => {
    // التحقق من المصادقة والصلاحيات عند تغيير الطريق
    const checkAuthorization = async () => {
      // إذا كانت الصفحة لا تتطلب مصادقة
      const publicPaths = ['/login', '/forgot-password'];
      if (publicPaths.includes(router.pathname)) {
        setAuthorized(true);
        return;
      }
      
      // التحقق من المصادقة
      if (!isAuthenticated) {
        setAuthorized(false);
        router.push('/login');
        return;
      }
      
      // التحقق من الصلاحية المطلوبة
      if (requiredPermission && currentAdmin) {
        const hasAccess = await hasPermission(currentAdmin, requiredPermission);
        setAuthorized(hasAccess);
        
        if (!hasAccess) {
          router.push('/unauthorized');
        }
      } else {
        setAuthorized(isAuthenticated);
      }
    };
    
    checkAuthorization();
  }, [router, isAuthenticated, currentAdmin, requiredPermission]);
  
  if (loading) {
    return <div>جاري التحميل...</div>;
  }
  
  if (!authorized) {
    return null;
  }
  
  return <>{children}</>;
}
```

## 7. توصيات للممارسات الجيدة

### 7.1 استخدام حراسة الصلاحيات بشكل متسق

- استخدم مكون `PermissionGuard` في كل مكان تحتاج فيه إلى حماية العناصر بالصلاحيات
- طبق حارس الطرق على مستوى التطبيق للتحقق من الصلاحيات للصفحات الكاملة
- استخدم حارس الطرق لتوجيه المستخدمين غير المصرح لهم إلى صفحات مناسبة

### 7.2 تعزيز أداء نظام الصلاحيات

- استخدم ذاكرة التخزين المؤقت (cache) للصلاحيات المستخدمة بشكل متكرر
- قم بتحميل الصلاحيات مرة واحدة عند تسجيل الدخول وتخزينها في Redux
- استخدم تحقق أولي في الواجهة قبل الاتصال بالخادم للتحقق النهائي

### 7.3 التوثيق والصيانة

- حافظ على سجل بجميع رموز الصلاحيات المستخدمة في التطبيق
- قم بتوثيق أي تغييرات في هيكل الصلاحيات
- اختبر النظام بشكل دوري للتأكد من عمله كما هو متوقع

### 8.2 هوك (Hook) مخصص للتحقق من الصلاحيات

```tsx
// في ملف src/domains/admins/hooks/usePermission.ts
import { useState, useEffect } from 'react';
import { useAppSelector } from '@/store/hooks';
import { hasPermission } from '../utils/permissions';

export function usePermission(permissionCode: string) {
  const currentAdmin = useAppSelector(state => state.auth.currentAdmin);
  const [loading, setLoading] = useState(true);
  const [hasAccess, setHasAccess] = useState(false);
  
  useEffect(() => {
    async function checkAccess() {
      setLoading(true);
      
      if (!currentAdmin) {
        setHasAccess(false);
        setLoading(false);
        return;
      }
      
      // التحقق الأولي من الصلاحيات المخزنة
      if (currentAdmin.permissions?.[permissionCode]) {
        setHasAccess(true);
        setLoading(false);
        return;
      }
      
      // التحقق من خلال واجهة برمجة التطبيق
      const result = await hasPermission(currentAdmin, permissionCode);
      setHasAccess(result);
      setLoading(false);
    }
    
    checkAccess();
  }, [currentAdmin, permissionCode]);
  
  return { loading, hasAccess };
}
```

## المحفزات والوظائف

### محفز إنشاء المستخدم

```sql
-- وظيفة لإضافة مستخدم جديد إلى جدول المسؤولين عند إنشاء حساب جديد
CREATE OR REPLACE FUNCTION public.process_new_admin_user()
RETURNS TRIGGER AS $$
DECLARE
  default_role_id uuid;
  admins_table_exists boolean;
BEGIN
  -- التحقق من وجود جدول المسؤولين
  SELECT EXISTS (
    SELECT FROM information_schema.tables 
    WHERE table_schema = 'public' AND table_name = 'admins'
  ) INTO admins_table_exists;
  
  IF NOT admins_table_exists THEN
    RETURN NEW; -- المستخدم تم إنشاؤه في auth.users لكن لا يوجد جدول admins
  END IF;

  -- الحصول على الدور الافتراضي (عادة 'viewer')
  SELECT id INTO default_role_id FROM public.roles WHERE name = 'viewer' LIMIT 1;
  
  -- إنشاء سجل جديد في جدول المسؤولين
  INSERT INTO public.admins (user_id, email, role_id)
  VALUES (NEW.id, NEW.email, default_role_id);
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- المحفز
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.process_new_admin_user();
```

### وظيفة التحقق من الصلاحيات

```sql
-- وظيفة محسنة للتحقق من صلاحيات المستخدم
CREATE OR REPLACE FUNCTION check_admin_permission_enhanced(
  admin_id uuid,
  permission_code text
) RETURNS boolean AS $$
DECLARE
  has_permission boolean;
  admin_record record;
  role_permissions jsonb;
  admin_permissions jsonb;
BEGIN
  -- جلب بيانات المسؤول
  SELECT a.*, r.permissions AS role_perms
  INTO admin_record
  FROM public.admins a
  LEFT JOIN public.roles r ON a.role_id = r.id
  WHERE a.id = admin_id;
  
  IF admin_record IS NULL THEN
    RETURN false;
  END IF;
  
  -- تحديد صلاحيات الدور
  role_permissions := admin_record.role_perms;
  
  -- تحديد الصلاحيات المخصصة للمسؤول (إن وجدت)
  admin_permissions := admin_record.permissions;
  
  -- التحقق من وجود الصلاحية
  -- أولاً في الصلاحيات المخصصة للمسؤول
  IF admin_permissions IS NOT NULL AND admin_permissions ? permission_code THEN
    has_permission := (admin_permissions ->> permission_code)::boolean;
  -- ثم في صلاحيات الدور
  ELSIF role_permissions IS NOT NULL AND role_permissions ? permission_code THEN
    has_permission := (role_permissions ->> permission_code)::boolean;
  ELSE
    has_permission := false;
  END IF;
  
  RETURN has_permission;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

## سياسات أمان Row Level Security (RLS)

```sql
-- سياسة للمسؤولين لمشاهدة بياناتهم الخاصة والمسؤولين الذين لديهم صلاحية إدارة
CREATE POLICY admins_policy
  ON public.admins
  FOR SELECT
  TO authenticated
  USING (user_id = auth.uid() OR EXISTS (
    SELECT 1 FROM public.admins a
    JOIN public.roles r ON a.role_id = r.id
    WHERE a.user_id = auth.uid()
    AND r.permissions->>'admins:manage' = 'true'
  ));

-- سياسة للأدوار للسماح بمشاهدتها للجميع
CREATE POLICY roles_select_policy
  ON public.roles
  FOR SELECT
  TO authenticated
  USING (true);

-- سياسة لحماية الصلاحيات للسماح بمشاهدتها للجميع
CREATE POLICY permissions_select_policy
  ON public.permissions
  FOR SELECT
  TO authenticated
  USING (true);
```

## دوال التحقق من الصلاحيات

### hasPermission

```typescript
// src/domains/admins/utils/permissions.ts

/**
 * التحقق من وجود صلاحية معينة لدى المسؤول
 */
export async function hasPermission(admin: Admin | null | undefined, permissionCode: string): Promise<boolean> {
  if (!admin) return false;
  
  // التحقق من الصلاحيات المخصصة للمسؤول
  if (admin.permissions && typeof admin.permissions[permissionCode] === 'boolean') {
    return admin.permissions[permissionCode] === true;
  }
  
  // التحقق من صلاحيات الدور
  if (admin.role && admin.role.permissions && typeof admin.role.permissions[permissionCode] === 'boolean') {
    return admin.role.permissions[permissionCode] === true;
  }
  
  // استخدام دالة RPC محسنة للتحقق من الصلاحيات
  try {
    const { data, error } = await supabase.rpc('check_admin_permission_enhanced', {
      admin_id: admin.id,
      permission_code: permissionCode
    });
    
    if (error) throw error;
    return !!data;
  } catch (error) {
    console.error('Error checking permission:', error);
    return false;
  }
}
```

## مكونات React الأساسية

### AuthProvider

```tsx
// src/domains/admins/providers/AuthProvider.tsx

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  const pathname = usePathname();
  const dispatch = useAppDispatch();
  const { isAuthenticated, loading } = useAppSelector((state) => state.auth);
  
  useEffect(() => {
    // التحقق من المصادقة
    dispatch(checkAuth());
  }, [dispatch]);
  
  useEffect(() => {
    // قائمة المسارات العامة
    const publicPaths = ['/login', '/register', '/forgot-password', '/reset-password'];
    const isPublicPath = publicPaths.some(path => pathname?.startsWith(path));
    
    // التوجيه بناءً على حالة المصادقة
    if (!isPublicPath && !loading && !isAuthenticated) {
      const returnUrl = encodeURIComponent(pathname || '/');
      router.push(`/login?returnUrl=${returnUrl}`);
    }
    
    // إذا كان المستخدم مصادقًا ويحاول الوصول إلى صفحة تسجيل الدخول
    if (isPublicPath && isAuthenticated) {
      router.push('/');
    }
  }, [isAuthenticated, loading, pathname, router]);
  
  return children;
}
```

### PermissionGuard

```tsx
// src/domains/admins/components/PermissionGuard.tsx

export function PermissionGuard({ 
  permissionCode, 
  children, 
  fallback = null 
}: PermissionGuardProps) {
  const { currentAdmin } = useAppSelector((state) => state.auth);
  const [hasAccess, setHasAccess] = useState(false);
  
  useEffect(() => {
    async function checkPermission() {
      const result = await hasPermission(currentAdmin, permissionCode);
      setHasAccess(result);
    }
    
    checkPermission();
  }, [currentAdmin, permissionCode]);
  
  return hasAccess ? children : fallback;
}
```

## 1. تسجيل المستخدمين الجدد

### A. نموذج التسجيل (React Component)

```tsx
// src/app/register/RegisterForm.tsx
import { useState } from 'react';
import { useRouter } from 'next/navigation';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { register, clearError } from '@/domains/admins/store/authSlice';
import { Button, Input, Alert } from '@/shared/components/ui';

export function RegisterForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');
  const [validationError, setValidationError] = useState<string | null>(null);
  
  const router = useRouter();
  const dispatch = useAppDispatch();
  const { loading, error, emailConfirmationRequired } = useAppSelector((state) => state.auth);
  
  // مسح أي أخطاء سابقة عند تحميل النموذج
  useEffect(() => {
    dispatch(clearError());
  }, [dispatch]);
  
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setValidationError(null);
    
    // التحقق من تطابق كلمتي المرور
    if (password !== confirmPassword) {
      setValidationError('كلمتا المرور غير متطابقتين');
      return;
    }
    
    // التحقق من طول كلمة المرور
    if (password.length < 8) {
      setValidationError('يجب أن تكون كلمة المرور 8 أحرف على الأقل');
      return;
    }
    
    // إرسال طلب التسجيل
    dispatch(register({
      name,
      email,
      password
    }));
  };
  
  // إذا تم التسجيل بنجاح وتم التأكيد، أو إذا كان التأكيد غير مطلوب
  if (emailConfirmationRequired) {
    return (
      <div className="text-center p-6 bg-green-50 rounded-lg">
        <h2 className="text-xl font-bold text-green-800 mb-2">تم التسجيل بنجاح!</h2>
        <p className="mb-4">تم إرسال رسالة تأكيد إلى بريدك الإلكتروني. يرجى التحقق من بريدك الإلكتروني وتأكيد حسابك.</p>
        <Button onClick={() => router.push('/login')}>العودة إلى تسجيل الدخول</Button>
      </div>
    );
  }
  
  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      {error && <Alert variant="destructive">{error}</Alert>}
      {validationError && <Alert variant="destructive">{validationError}</Alert>}
      
      <div>
        <label htmlFor="name" className="block text-sm font-medium text-gray-700">الاسم</label>
        <Input 
          id="name" 
          type="text" 
          value={name} 
          onChange={(e) => setName(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <label htmlFor="email" className="block text-sm font-medium text-gray-700">البريد الإلكتروني</label>
        <Input 
          id="email" 
          type="email" 
          value={email} 
          onChange={(e) => setEmail(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <label htmlFor="password" className="block text-sm font-medium text-gray-700">كلمة المرور</label>
        <Input 
          id="password" 
          type="password" 
          value={password} 
          onChange={(e) => setPassword(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <label htmlFor="confirmPassword" className="block text-sm font-medium text-gray-700">تأكيد كلمة المرور</label>
        <Input 
          id="confirmPassword" 
          type="password" 
          value={confirmPassword} 
          onChange={(e) => setConfirmPassword(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <Button type="submit" className="w-full" disabled={loading}>
          {loading ? 'جاري التسجيل...' : 'تسجيل'}
        </Button>
      </div>
    </form>
  );
}
```

```tsx
// src/app/login/LoginForm.tsx
import { useState, useEffect } from 'react';
import { useRouter, useSearchParams } from 'next/navigation';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { login, clearError } from '@/domains/admins/store/authSlice';
import { Button, Input, Alert } from '@/shared/components/ui';
import Link from 'next/link';

export function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  
  const router = useRouter();
  const searchParams = useSearchParams();
  const returnUrl = searchParams?.get('returnUrl') || '/';
  
  const dispatch = useAppDispatch();

    e.preventDefault();
```

```typescript
// src/domains/admins/hooks/usePermission.ts
import { useState, useEffect } from 'react';
import { useAppSelector } from '@/store/hooks';
import { hasPermission } from '@/domains/admins/utils/permissions';

export function usePermission(permissionCode: string) {
  const { currentAdmin } = useAppSelector((state) => state.auth);
  const [hasAccess, setHasAccess] = useState(false);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    async function checkPermission() {
      setLoading(true);
      try {
        const result = await hasPermission(currentAdmin, permissionCode);
        setHasAccess(result);
      } catch (error) {
        console.error('خطأ في التحقق من الصلاحية:', error);
        setHasAccess(false);
      } finally {
        setLoading(false);
      }
    }
    
    checkPermission();
  }, [currentAdmin, permissionCode]);
  
  return { hasAccess, loading };
}

// استخدام الـ hook
function AdminComponent() {

  if (loading) return <p>جاري التحميل...</p>;

        <button className="btn btn-primary">إنشاء طلب جديد</button>
      ) : (
        <p className="text-red-500">ليس لديك صلاحية لإنشاء طلب جديد</p>
      )}
    </div>
  );
}
```

```tsx
import { ReactNode, useState, useEffect } from 'react';
import { useAppSelector } from '@/store/hooks';
import { hasPermission } from '@/domains/admins/utils/permissions';

interface PermissionGuardProps {
  permissionCode: string;
  children: ReactNode;
  fallback?: ReactNode;
}

export function PermissionGuard({ permissionCode, children, fallback = null }: PermissionGuardProps) {
  const { currentAdmin } = useAppSelector((state) => state.auth);
  const [hasAccess, setHasAccess] = useState(false);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    async function checkPermission() {
      setLoading(true);
      try {
        const result = await hasPermission(currentAdmin, permissionCode);
        setHasAccess(result);
      } catch (error) {
        console.error('خطأ في التحقق من الصلاحية:', error);
        setHasAccess(false);
      } finally {
        setLoading(false);
      }
    }
    
    checkPermission();
  }, [currentAdmin, permissionCode]);
  
  if (loading) return null;
  
  return hasAccess ? <>{children}</> : <>{fallback}</>;
}
```

```tsx
// src/domains/admins/providers/AuthProvider.tsx
import { useEffect, ReactNode } from 'react';
import { usePathname, useRouter } from 'next/navigation';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { checkAuth } from '@/domains/admins/store/authSlice';

// قائمة المسارات العامة التي لا تتطلب مصادقة
const PUBLIC_PATHS = [
  '/login',
  '/register',
  '/forgot-password',
  '/reset-password',
];

export function AuthProvider({ children }: { children: ReactNode }) {
  const router = useRouter();
  const pathname = usePathname();
  const dispatch = useAppDispatch();
  const { isAuthenticated, loading } = useAppSelector((state) => state.auth);
  
  useEffect(() => {
    // التحقق من المصادقة عند كل تحميل للتطبيق
    dispatch(checkAuth());
  }, [dispatch]);
  
  useEffect(() => {
    // التحقق مما إذا كان المسار الحالي عامًا أم محميًا
    const isPublicPath = PUBLIC_PATHS.some(path => pathname?.startsWith(path));
    
    // إذا كان المستخدم غير مصادق ويحاول الوصول إلى مسار محمي
    if (!isPublicPath && !loading && !isAuthenticated) {
      const returnUrl = encodeURIComponent(pathname || '/');
      router.push(`/login?returnUrl=${returnUrl}`);
    }
    
    // إذا كان المستخدم مصادقًا ويحاول الوصول إلى صفحة تسجيل الدخول أو التسجيل
    if (isPublicPath && isAuthenticated) {
      router.push('/');
    }
  }, [isAuthenticated, loading, pathname, router]);
  
  return <>{children}</>;
}
```

```tsx
'use client';

import { useState, useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { useAppSelector } from '@/store/hooks';
import { hasPermission } from '@/domains/admins/utils/permissions';

export default function AdminUsersPage() {
  const router = useRouter();
  const { currentAdmin, isAuthenticated } = useAppSelector((state) => state.auth);

  const [loading, setLoading] = useState(true);

    async function checkAuthorization() {
      setLoading(true);
      
      if (!isAuthenticated) {
        router.push('/login?returnUrl=/admin/users');
        return;

      try {
        // التحقق من صلاحية إدارة المستخدمين
        const canManageUsers = await hasPermission(currentAdmin, 'admins:manage');
        
        if (canManageUsers) {
          setAuthorized(true);
        } else {
          // توجيه إلى صفحة غير مصرح
          router.push('/unauthorized');
        }
      } catch (error) {
        console.error('خطأ في التحقق من الصلاحية:', error);
        router.push('/unauthorized');
      } finally {
        setLoading(false);
      }
    }
    
    checkAuthorization();
  }, [currentAdmin, isAuthenticated, router]);
  
  if (loading) {
    return (
      <div className="flex justify-center items-center min-h-screen">
        <div className="text-center">
          <div className="w-16 h-16 border-4 border-t-blue-500 border-gray-200 rounded-full animate-spin mx-auto"></div>
          <p className="mt-4 text-gray-600">جاري التحميل...</p>
        </div>
      </div>
    );
  }
  
  if (!authorized) {
    return null; // سيتم توجيه المستخدم، لذا لا داعي لعرض أي محتوى
  }
  
  return (
    <div className="container mx-auto py-8">
      <h1 className="text-2xl font-bold mb-6">إدارة المستخدمين</h1>
      {/* محتوى الصفحة */}

}
```

### B. نموذج تعيين كلمة مرور جديدة

```tsx
// src/app/reset-password/[token]/ResetPasswordForm.tsx
import { useState, useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { updatePassword, clearError } from '@/domains/admins/store/authSlice';
import { Button, Input, Alert } from '@/shared/components/ui';

interface ResetPasswordFormProps {
  token: string;
}

export function ResetPasswordForm({ token }: ResetPasswordFormProps) {
  const [password, setPassword] = useState('');
  const [confirmPassword, setConfirmPassword] = useState('');
  const [validationError, setValidationError] = useState<string | null>(null);
  const [success, setSuccess] = useState(false);
  
  const router = useRouter();
  const dispatch = useAppDispatch();
  const { loading, error } = useAppSelector((state) => state.auth);
  
  // مسح أي أخطاء سابقة عند تحميل النموذج
  useEffect(() => {
    dispatch(clearError());
  }, [dispatch]);
  
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setValidationError(null);
    
    // التحقق من تطابق كلمتي المرور
    if (password !== confirmPassword) {
      setValidationError('كلمتا المرور غير متطابقتين');
      return;
    }
    
    // التحقق من طول كلمة المرور
    if (password.length < 8) {
      setValidationError('يجب أن تكون كلمة المرور 8 أحرف على الأقل');
      return;
    }
    
    // إرسال طلب تحديث كلمة المرور
    const result = await dispatch(updatePassword({
      token,
      password
    })).unwrap();
    
    if (result.success) {
      setSuccess(true);
      
      // توجيه المستخدم إلى صفحة تسجيل الدخول بعد 3 ثوانٍ
      setTimeout(() => {
        router.push('/login');
      }, 3000);
    }
  };
  
  if (success) {
    return (
      <div className="text-center p-6 bg-green-50 rounded-lg">
        <h2 className="text-xl font-bold text-green-800 mb-2">تم تحديث كلمة المرور بنجاح</h2>
        <p className="mb-4">سيتم توجيهك إلى صفحة تسجيل الدخول...</p>
      </div>
    );
  }
  
  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      {error && <Alert variant="destructive">{error}</Alert>}
      {validationError && <Alert variant="destructive">{validationError}</Alert>}
      
      <div>
        <label htmlFor="password" className="block text-sm font-medium text-gray-700">كلمة المرور الجديدة</label>
        <Input 
          id="password" 
          type="password" 
          value={password} 
          onChange={(e) => setPassword(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <label htmlFor="confirmPassword" className="block text-sm font-medium text-gray-700">تأكيد كلمة المرور</label>
        <Input 
          id="confirmPassword" 
          type="password" 
          value={confirmPassword} 
          onChange={(e) => setConfirmPassword(e.target.value)} 
          required 
        />
      </div>
      
      <div>
        <Button type="submit" className="w-full" disabled={loading}>
          {loading ? 'جاري الحفظ...' : 'تحديث كلمة المرور'}
        </Button>
      </div>
    </form>
  );
}
```

## 9. تعديل الصلاحيات

### A. مكون إدارة صلاحيات المستخدم

```tsx
// src/domains/admins/components/AdminPermissionsManager.tsx
import { useState, useEffect } from 'react';
import { useAppDispatch, useAppSelector } from '@/store/hooks';
import { updateAdminPermissions } from '@/domains/admins/store/authSlice';
import { Switch, Button, Alert } from '@/shared/components/ui';
import { hasPermission } from '@/domains/admins/utils/permissions';

interface AdminPermissionsManagerProps {
  adminId: string;
}

export function AdminPermissionsManager({ adminId }: AdminPermissionsManagerProps) {
  const [permissions, setPermissions] = useState<Record<string, boolean>>({});
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [success, setSuccess] = useState(false);
  
  const dispatch = useAppDispatch();
  const { currentAdmin } = useAppSelector((state) => state.auth);
  
  // التحقق من صلاحية إدارة المستخدمين
  const [canManageUsers, setCanManageUsers] = useState(false);
  
  useEffect(() => {
    async function checkPermission() {
      const result = await hasPermission(currentAdmin, 'admins:manage');
      setCanManageUsers(result);
    }
    
    checkPermission();
  }, [currentAdmin]);
  
  // جلب صلاحيات المستخدم الحالية
  useEffect(() => {
    async function fetchAdminPermissions() {
      setLoading(true);
      setError(null);
      
      try {
        const response = await fetch(`/api/admins/${adminId}/permissions`);
        
        if (!response.ok) {
          throw new Error('فشل جلب صلاحيات المستخدم');
        }
        
        const data = await response.json();
        setPermissions(data.permissions || {});
      } catch (err) {
        setError(err instanceof Error ? err.message : 'حدث خطأ أثناء جلب الصلاحيات');
      } finally {
        setLoading(false);
      }
    }
    
    if (canManageUsers) {
      fetchAdminPermissions();
    }
  }, [adminId, canManageUsers]);
  
  const handlePermissionChange = (permissionCode: string, value: boolean) => {
    setPermissions(prev => ({
      ...prev,
      [permissionCode]: value
    }));
  };
  
  const handleSave = async () => {
    setError(null);
    setSuccess(false);
    
    try {
      const response = await fetch(`/api/admins/${adminId}/permissions`, {
        method: 'PATCH',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ permissions })
      });
      
      if (!response.ok) {
        throw new Error('فشل تحديث صلاحيات المستخدم');
      }
      
      setSuccess(true);
      
      // تحديث الصلاحيات في Redux إذا كان المستخدم الحالي هو المسؤول الحالي
      if (currentAdmin && currentAdmin.id === adminId) {
        dispatch(updateAdminPermissions(permissions));
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'حدث خطأ أثناء تحديث الصلاحيات');
    }
  };
  
  if (loading) {
    return <p>جاري تحميل الصلاحيات...</p>;
  }
  
  if (!canManageUsers) {
    return <p className="text-red-500">ليس لديك صلاحية لإدارة المستخدمين</p>;
  }
  
  return (
    <div className="space-y-6">
      <h2 className="text-xl font-bold">إدارة صلاحيات المستخدم</h2>
      
      {error && <Alert variant="destructive">{error}</Alert>}
      {success && <Alert variant="success">تم تحديث الصلاحيات بنجاح</Alert>}
      
      <div className="space-y-4">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
          {/* المكونات المختلفة للصلاحيات */}
          <div className="p-4 border rounded-md">
            <h3 className="font-medium mb-2">الطلبات</h3>
            <div className="space-y-2">
              <div className="flex justify-between items-center">
                <label className="text-sm">عرض الطلبات</label>
                <Switch 
                  checked={permissions['orders:view'] || false} 
                  onCheckedChange={(value) => handlePermissionChange('orders:view', value)} 
                />
              </div>
              <div className="flex justify-between items-center">
                <label className="text-sm">إنشاء طلبات</label>
                <Switch 
                  checked={permissions['orders:create'] || false} 
                  onCheckedChange={(value) => handlePermissionChange('orders:create', value)} 
                />
              </div>
              <div className="flex justify-between items-center">
                <label className="text-sm">تعديل الطلبات</label>
                <Switch 
                  checked={permissions['orders:edit'] || false} 
                  onCheckedChange={(value) => handlePermissionChange('orders:edit', value)} 
                />
              </div>
              <div className="flex justify-between items-center">
                <label className="text-sm">حذف الطلبات</label>
                <Switch 
                  checked={permissions['orders:delete'] || false} 
                  onCheckedChange={(value) => handlePermissionChange('orders:delete', value)} 
                />
              </div>
            </div>
          </div>
          
          {/* يمكن إضافة المزيد من مجموعات الصلاحيات هنا */}
        </div>
        
        <div className="flex justify-end">
          <Button onClick={handleSave}>حفظ الصلاحيات</Button>
        </div>
      </div>
    </div>
  );
}
```
